   362→
   363→  // ═══════════════════════════════════════════════════════
   364→  // getMiniStepStatus — Permission-Driven with Business Rules
   365→  // Permissions are the source of truth for WHO can do WHAT.
   366→  // Business rules define WHEN things are accessible:
   367→  //   - INTER-S progression: S2 requires S1 completed, S3 requires S2, etc.
   368→  //   - INTRA-S progression: step N requires step N-1 completed
   369→  //   - Step 5 (audit): requires steps 1-4 ALL completed within the same S
   370→  //   - a1 (perform) → can enter and act (if previous step done)
   371→  //   - a0 (view) only → can see the step exists, but CANNOT enter it
   372→  //   - No permission → locked (not even visible)
   373→  //   - IMPORTANT: Previous step completion is checked at ZONE level,
   374→  //     so an auditor can access step 5 if employees completed 1-4
   375→  //   - For step 1 (individual): also checks EmployeeProgress for the
   376→  //     current user, so the employee's own completion unlocks step 2
   377→  // ═══════════════════════════════════════════════════════
   378→  getMiniStepStatus: (sStep, miniStep) => {
   379→    const { progress, employeeProgress, currentUser, adminFreeNavigation, currentZone } = get()
   380→    if (!currentUser) return 'locked'
   381→
   382→    const canViewStep = get().canView(sStep, miniStep)
   383→    const canPerformStep = get().canPerform(sStep, miniStep)
   384→    // Permission-driven: skip_steps from DB replaces hardcoded isAdmin check
   385→    const canSkipSteps = get().hasPermission('skip_steps')
   386→    const skipLocks = canSkipSteps && adminFreeNavigation
   387→
   388→    // Check if a specific mini-step of a specific S is completed at zone level
   389→    const isStepCompletedAt = (s: number, ms: number): boolean => {
   390→      if (currentZone) {
   391→        const zoneStep = progress.find(p =>
   392→          p.sStep === s &&
   393→          p.miniStep === ms &&
   394→          (p.zoneId === currentZone.id || p.zoneId === null) &&
   395→          p.completed
   396→        )
   397→        return !!zoneStep
   398→      }
   399→      const anyStep = progress.find(p =>
   400→        p.sStep === s &&
   401→        p.miniStep === ms &&
   402→        p.completed
   403→      )
   404→      return !!anyStep
   405→    }
   406→
   407→    // Check if a specific S-step (all 5 mini-steps) is fully completed (quesito earned)
   408→    // Also checks employeeProgress for individual steps, consistent with areSteps1to4Completed
   409→    const isSCompleted = (s: number): boolean => {
   410→      for (let ms = 1; ms <= 5; ms++) {
   411→        // Check zone-level progress first
   412→        if (isStepCompletedAt(s, ms)) continue
   413→        // Also check employeeProgress for this step (e.g., individual formación step 1)
   414→        const anyEmpCompleted = currentZone
   415→          ? employeeProgress.some(ep =>
   416→              ep.sStep === s &&
   417→              ep.miniStep === ms &&
   418→              ep.zoneId === currentZone.id &&
   419→              ep.completed
   420→            )
   421→          : employeeProgress.some(ep =>
   422→              ep.sStep === s &&
   423→              ep.miniStep === ms &&
   424→              ep.completed
   425→            )
   426→        if (!anyEmpCompleted) return false
   427→      }
   428→      return true
   429→    }
   430→
   431→    // Check if the CURRENT USER has completed step 1 individually (EmployeeProgress)
   432→    // This is crucial for unlocking step 2 for an employee who passed the exam
   433→    // even if not ALL employees in the zone have completed it yet
   434→    const hasUserCompletedStep1 = (): boolean => {
   435→      if (!currentZone) return false
   436→      const userStep = employeeProgress.find(ep =>
   437→        ep.sStep === sStep &&
   438→        ep.miniStep === 1 &&
   439→        ep.zoneId === currentZone.id &&
   440→        ep.userId === currentUser.id &&
   441→        ep.completed
   442→      )
   443→      return !!userStep
   444→    }
   445→
   446→    // Check if already completed at zone level (or project level if no zone)
   447→    const isStepCompleted = (): boolean => isStepCompletedAt(sStep, miniStep)
   448→
   449→    // Check if the CURRENT USER has completed this individual step (EmployeeProgress)
   450→    // For step 1 (formación) and step 4 (autoevaluación), completion is tracked per employee
   451→    const hasUserCompletedIndividualStep = (): boolean => {
   452→      if (!currentZone) return false
   453→      if (miniStep !== 1 && miniStep !== 4) return false // Only individual steps
   454→      const userStep = employeeProgress.find(ep =>
   455→        ep.sStep === sStep &&
   456→        ep.miniStep === miniStep &&
   457→        ep.zoneId === currentZone.id &&
   458→        ep.userId === currentUser.id &&
   459→        ep.completed
   460→      )
   461→      return !!userStep
   462→    }
   463→
   464→    // Combined: step is "done" if either zone-level completed OR user completed individually
   465→    const isStepDoneForUser = (): boolean => isStepCompleted() || hasUserCompletedIndividualStep()
   466→
   467→    // Check if step is completed by ANYONE (zone or any employee)
   468→    // This is crucial: ALL users must see completed steps in green,
   469→    // even if they don't have permission to enter them (e.g., auditors viewing steps 1-4)
   470→    const isStepCompletedByAnyone = (): boolean => {
   471→      if (isStepCompleted()) return true
   472→      if (!currentZone) {
   473→        return employeeProgress.some(ep =>
   474→          ep.sStep === sStep &&
   475→          ep.miniStep === miniStep &&
   476→          ep.completed
   477→        )
   478→      }
   479→      return employeeProgress.some(ep =>
   480→        ep.sStep === sStep &&
   481→        ep.miniStep === miniStep &&
   482→        ep.zoneId === currentZone.id &&
   483→        ep.completed
   484→      )
   485→    }
   486→
   487→    // Helper: check if steps 1-4 are all completed for this S-step
   488→    // Uses BOTH zone-level progress AND any employee progress,
   489→    // so auditors can access step 5 when employees have completed 1-4
   490→    const areSteps1to4Completed = (): boolean => {
   491→      for (let ms = 1; ms <= 4; ms++) {
   492→        const zoneCompleted = isStepCompletedAt(sStep, ms)
   493→        if (zoneCompleted) continue
   494→        // Check if ANY employee has completed this step individually
   495→        const anyEmpCompleted = currentZone
   496→          ? employeeProgress.some(ep =>
   497→              ep.sStep === sStep &&
   498→              ep.miniStep === ms &&
   499→              ep.zoneId === currentZone.id &&
   500→              ep.completed
   501→            )
   502→          : employeeProgress.some(ep =>
   503→              ep.sStep === sStep &&
   504→              ep.miniStep === ms &&
   505→              ep.completed
   506→            )
   507→        if (!anyEmpCompleted) return false
   508→      }
   509→      return true
   510→    }
   511→
   512→    // ═══ INTER-S PROGRESSION CHECK ═══
   513→    // S1 is always available. S2 requires S1 completed. S3 requires S2. Etc.
   514→    // This enforces the 5S methodology sequence: Clasificar → Ordenar → Limpiar → Estandarizar → Mantener
   515→    const isPreviousSCompleted = (): boolean => {
   516→      if (sStep === 1) return true // S1 has no prerequisite
   517→      return isSCompleted(sStep - 1)
   518→    }
   519→
   520→    // ═══ INTRA-S PROGRESSION CHECK ═══
   521→    // Step 1: always available (if you have permission AND previous S is done)
   522→    // Steps 2-4: need the immediately previous step completed
   523→    // Step 5: needs ALL steps 1-4 completed
   524→    // IMPORTANT: Completion checks include BOTH zone-level AND any employee progress,
   525→    // so auditors/responsables can see steps as available when employees completed them
   526→    const isPreviousStepCompleted = (): boolean => {
   527→      if (miniStep === 1) return true
   528→      if (miniStep === 5) return areSteps1to4Completed()
   529→      // Steps 2, 3, 4: need previous step completed at zone level
   530→      if (isStepCompletedAt(sStep, miniStep - 1)) return true
   531→      // For step 2: current user completed step 1 individually (exam passed)
   532→      if (miniStep === 2 && hasUserCompletedStep1()) return true
   533→      // Also check: ANY employee completed previous step (for auditors/responsables)
   534→      const anyEmpCompletedPrev = currentZone
   535→        ? employeeProgress.some(ep =>
   536→            ep.sStep === sStep &&
   537→            ep.miniStep === miniStep - 1 &&
   538→            ep.zoneId === currentZone.id &&
   539→            ep.completed
   540→          )
   541→        : employeeProgress.some(ep =>
   542→            ep.sStep === sStep &&
   543→            ep.miniStep === miniStep - 1 &&
   544→            ep.completed
   545→          )
   546→      if (anyEmpCompletedPrev) return true
   547→      return false
   548→    }
   549→
   550→    // ── If already completed (zone-level OR user's individual), VERIFY COHERENCY ──
   551→    if (isStepDoneForUser()) {
   552→      const isChainCoherent = (): boolean => {
   553→        // First check inter-S: previous S must be completed
   554→        if (!isPreviousSCompleted()) return false
   555→        // Then check intra-S chain — reuse same logic as isPreviousStepCompleted
   556→        if (miniStep === 1) return true // Step 1 is always coherent if previous S is done
   557→        if (miniStep === 5) return areSteps1to4Completed()
   558→        if (isStepCompletedAt(sStep, miniStep - 1)) return true
   559→        if (miniStep === 2 && hasUserCompletedStep1()) return true
   560→        // Also check any employee completed previous step
   561→        const anyEmpPrev = currentZone
   562→          ? employeeProgress.some(ep =>
   563→              ep.sStep === sStep &&
   564→              ep.miniStep === miniStep - 1 &&
   565→              ep.zoneId === currentZone.id &&
   566→              ep.completed
   567→            )
   568→          : employeeProgress.some(ep =>
   569→              ep.sStep === sStep &&
   570→              ep.miniStep === miniStep - 1 &&
   571→              ep.completed
   572→            )
   573→        if (anyEmpPrev) return true
   574→        return false
   575→      }
   576→
   577→      if (isChainCoherent()) {
   578→        if (canPerformStep || canSkipSteps) return 'completed'
   579→        if (canViewStep) return 'completed_viewonly'
   580→        return 'locked'
   581→      }
   582→      // Chain is incoherent: treat as NOT completed — fall through
   583→    }
   584→
   585→    // ── Step completed by ANYONE (zone or employee) → show as green for ALL users ──
   586→    // Even users without permission must see completed steps in green
   587→    if (isStepCompletedByAnyone()) {
   588→      if (canPerformStep || canSkipSteps) return 'completed'
   589→      // Everyone else sees it as completed_viewonly (green, read-only)
   590→      return 'completed_viewonly'
   591→    }
   592→
   593→    // ── Admin with lock open: skip all checks ──
   594→    if (skipLocks) return 'available'
   595→
   596→    // ── No permission at all = locked ──
   597→    if (!canViewStep && !canPerformStep) return 'locked'
   598→
   599→    // ── Has perform permission (a1) → check progression ──
   600→    if (canPerformStep) {
   601→      // INTER-S: can't start this S until previous S is completed
   602→      if (!isPreviousSCompleted()) return 'locked'
   603→      // INTRA-S: can't enter step until previous step is completed
   604→      if (!isPreviousStepCompleted()) return 'locked'
   605→      return 'available'
   606→    }
   607→
   608→    // ── Has view permission only (a0) → can see but CANNOT enter ──
   609→    if (canViewStep) return 'locked'
   610→
   611→    return 'locked'
   612→  },
   613→
   614→  isZoneMiniStepComplete: (sStep, miniStep, zoneId) => {
   615→    const { progress, employeeProgress, currentProject } = get()
   616→    if (!currentProject) return false
   617→
   618→    if (ZONE_MINI_STEPS.includes(miniStep)) {
   619→      // Zone step: check if the zone's Progress record is completed
   620→      const zoneStep = progress.find(p =>
   621→        p.sStep === sStep &&
   622→        p.miniStep === miniStep &&
   623→        p.zoneId === zoneId
   624→      )
   625→      return zoneStep?.completed ?? false
   626→    }
   627→
   628→    if (INDIVIDUAL_MINI_STEPS.includes(miniStep)) {
   629→      // Individual step (1 only): ALL employees in the zone must have completed
   630→      // The zone-level Progress record for step 1 is only marked completed when ALL employees pass
   631→      const zoneStep = progress.find(p =>
   632→        p.sStep === sStep &&
   633→        p.miniStep === miniStep &&
   634→        (p.zoneId === zoneId || p.zoneId === null) &&
   635→        p.completed
   636→      )
   637→      return !!zoneStep
   638→    }
   639→
   640→    return false
   641→  },
   642→
   643→  isQuesitoEarned: (sStep) => {
   644→    const { progress, employeeProgress, currentZone } = get()
   645→    const zoneId = currentZone?.id || null
   646→
   647→    // Check ALL 5 mini-steps are completed — each one individually
   648→    // For zone-level steps (2,3,5): check progress table
   649→    // For individual steps (1,4): also check employeeProgress as fallback
   650→    for (let miniStep = 1; miniStep <= 5; miniStep++) {
   651→      // Check zone-level progress first
   652→      const zoneStep = progress.find(p =>
   653→        p.sStep === sStep &&
   654→        p.miniStep === miniStep &&
   655→        (p.zoneId === zoneId || p.zoneId === null) &&
   656→        p.completed
   657→      )
   658→      if (zoneStep) continue
   659→
   660→      // Also check employeeProgress for this step (e.g., individual formación step 1)
   661→      // This ensures the quesito is earned even when not all employees have completed step 1,
   662→      // but the auditor was able to pass step 5 (audit) because areSteps1to4Completed returned true
   663→      const anyEmpCompleted = currentZone
   664→        ? employeeProgress.some(ep =>
   665→            ep.sStep === sStep &&
   666→            ep.miniStep === miniStep &&
   667→            ep.zoneId === currentZone.id &&
   668→            ep.completed
   669→          )
   670→        : employeeProgress.some(ep =>
   671→            ep.sStep === sStep &&
   672→            ep.miniStep === miniStep &&
   673→            ep.completed
   674→          )
   675→      if (anyEmpCompleted) continue
   676→
   677→      return false
   678→    }
   679→    return true
   680→  },
   681→
   682→  is5SCompleted: () => {
   683→    for (let i = 1; i <= 5; i++) {
   684→      if (!get().isQuesitoEarned(i)) return false
   685→    }
   686→    return true
   687→  },
   688→
   689→  getCompletedCount: () => {
   690→    const { progress, currentZone } = get()
   691→
   692→    if (currentZone) {
   693→      const completedMiniSteps = progress.filter(p =>
   694→        (p.zoneId === currentZone.id || p.zoneId === null) &&
   695→        p.completed
   696→      ).length
   697→      let completedSSteps = 0
   698→      for (let i = 1; i <= 5; i++) {
   699→        if (get().isQuesitoEarned(i)) completedSSteps++
   700→      }
   701→      return { sSteps: completedSSteps, miniSteps: completedMiniSteps, total: 25 }
   702→    }
   703→
   704→    const completedMiniSteps = progress.filter(p => p.completed).length
   705→    let completedSSteps = 0
   706→    for (let i = 1; i <= 5; i++) {
   707→      if (get().isQuesitoEarned(i)) completedSSteps++
   708→    }
   709→    return { sSteps: completedSSteps, miniSteps: completedMiniSteps, total: 25 }
   710→  },

   712→  // Auth & Project Actions
   713→  login: async (email: string, password: string) => {
   714→    try {
   715→      set({ isLoginLoading: true, authError: null })
   716→      const res = await fetch('/api/auth', {
   717→        method: 'POST',
   718→        headers: { 'Content-Type': 'application/json' },
   719→        body: JSON.stringify({ email, password }),
   720→      })
   721→      const data = await res.json()
   722→
   723→      if (!res.ok) {
   724→        const errorMsg = data.error || 'Email o contraseña incorrectos'
   725→        set({ isLoginLoading: false, authError: errorMsg })
   726→        return false
   727→      }
   728→
   729→      if (!data.user) {
   730→        set({ isLoginLoading: false, authError: 'Error inesperado al iniciar sesión' })
   731→        return false
   732→      }
   733→
   734→      set({ currentUser: data.user, isLoginLoading: false, authError: null })
   735→
   736→      // Load permissions after login
   737→      try {
   738→        await get().fetchPermissions()
   739→      } catch (e) {
   740→        console.error('Error loading permissions after login:', e)
   741→      }
   742→
   743→      // Gestor (dueño de la app) goes directly to management panel
   744→      if (data.user.role === 'gestor') {
   745→        set({ authView: 'board', activeTab: 'gestion' })
   746→        try {
   747→          await get().fetchCompanies()
   748→        } catch (e) {
   749→          console.error('Error fetching companies after login:', e)
   750→        }
   751→        return true
   752→      }
   753→
   754→      // Check for projects after login
   755→      try {
   756→        await get().fetchProjects()
   757→      } catch (e) {
   758→        console.error('Error fetching projects after login:', e)
   759→      }
   760→      const { projects } = get()
   761→
   762→      if (projects.length > 0) {
   763→        set({ currentProject: projects[0], authView: 'board' })
   764→        // Fetch user's assigned zones after login
   765→        try {
   766→          await get().fetchUserZones()
   767→        } catch (e) {
   768→          console.error('Error fetching user zones after login:', e)
   769→        }
   770→      } else {
   771→        // Only admin can create projects via setup wizard
   772→        // Non-admin users see a waiting screen
   773→        if (data.user.role === 'admin') {
   774→          set({ authView: 'setup' })
   775→        } else {
   776→          set({ authView: 'no_projects' })
   777→        }
   778→      }
   779→
   780→      return true
   781→    } catch (error) {
   782→      console.error('Login error:', error)
   783→      set({ isLoginLoading: false, authError: 'Error de conexión. Inténtalo de nuevo.' })
   784→      return false
   785→    }
   786→  },
