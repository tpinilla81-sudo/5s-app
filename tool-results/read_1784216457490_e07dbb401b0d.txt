   299‚Üí   299‚Üí    }

   303‚Üí   303‚Üí      const userZoneIds = userZones.map(uz => uz.id)
   304‚Üí   304‚Üí      const assignedZones = currentProject.zones.filter(z => userZoneIds.includes(z.id))

   310‚Üí   310‚Üí
   311‚Üí   311‚Üí    // Fallback: only assigned zones
   312‚Üí   312‚Üí    const userZoneIds = userZones.map(uz => uz.id)
   313‚Üí   313‚Üí    return currentProject.zones.filter(z => userZoneIds.includes(z.id))

   360‚Üí   360‚Üí  // ‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê
   361‚Üí   361‚Üí  // getMiniStepStatus ‚Äî Permission-Driven with Business Rules
   362‚Üí   362‚Üí  // Permissions are the source of truth for WHO can do WHAT.
   363‚Üí   363‚Üí  // Business rules define WHEN things are accessible:
   364‚Üí   364‚Üí  //   - INTER-S progression: S2 requires S1 completed, S3 requires S2, etc.
   365‚Üí   365‚Üí  //   - INTRA-S progression: step N requires step N-1 completed
   366‚Üí   366‚Üí  //   - Step 5 (audit): requires steps 1-4 ALL completed within the same S
   367‚Üí   367‚Üí  //   - a1 (perform) ‚Üí can enter and act (if previous step done)
   368‚Üí   368‚Üí  //   - a0 (view) only ‚Üí can see the step exists, but CANNOT enter it
   369‚Üí   369‚Üí  //   - No permission ‚Üí locked (not even visible)
   370‚Üí   370‚Üí  //   - IMPORTANT: Previous step completion is checked at ZONE level,
   371‚Üí   371‚Üí  //     so an auditor can access step 5 if employees completed 1-4
   372‚Üí   372‚Üí  //   - For step 1 (individual): also checks EmployeeProgress for the
   373‚Üí   373‚Üí  //     current user, so the employee's own completion unlocks step 2
   374‚Üí   374‚Üí  // ‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê
   375‚Üí   375‚Üí  getMiniStepStatus: (sStep, miniStep) => {
   376‚Üí   376‚Üí    const { progress, employeeProgress, currentUser, adminFreeNavigation, currentZone } = get()
   377‚Üí   377‚Üí    if (!currentUser) return 'locked'
   378‚Üí   378‚Üí
   379‚Üí   379‚Üí    const canViewStep = get().canView(sStep, miniStep)
   380‚Üí   380‚Üí    const canPerformStep = get().canPerform(sStep, miniStep)
   381‚Üí   381‚Üí    // Permission-driven: skip_steps from DB replaces hardcoded isAdmin check
   382‚Üí   382‚Üí    const canSkipSteps = get().hasPermission('skip_steps')
   383‚Üí   383‚Üí    const skipLocks = canSkipSteps && adminFreeNavigation
   384‚Üí   384‚Üí
   385‚Üí   385‚Üí    // Check if a specific mini-step of a specific S is completed at zone level
   386‚Üí   386‚Üí    const isStepCompletedAt = (s: number, ms: number): boolean => {
   387‚Üí   387‚Üí      if (currentZone) {
   388‚Üí   388‚Üí        const zoneStep = progress.find(p =>
   389‚Üí   389‚Üí          p.sStep === s &&
   390‚Üí   390‚Üí          p.miniStep === ms &&
   391‚Üí   391‚Üí          (p.zoneId === currentZone.id || p.zoneId === null) &&
   392‚Üí   392‚Üí          p.completed
   393‚Üí   393‚Üí        )
   394‚Üí   394‚Üí        return !!zoneStep
   395‚Üí   395‚Üí      }
   396‚Üí   396‚Üí      const anyStep = progress.find(p =>
   397‚Üí   397‚Üí        p.sStep === s &&
   398‚Üí   398‚Üí        p.miniStep === ms &&
   399‚Üí   399‚Üí        p.completed
   400‚Üí   400‚Üí      )
   401‚Üí   401‚Üí      return !!anyStep
   402‚Üí   402‚Üí    }
   403‚Üí   403‚Üí
   404‚Üí   404‚Üí    // Check if a specific S-step (all 5 mini-steps) is fully completed (quesito earned)
   405‚Üí   405‚Üí    // Also checks employeeProgress for individual steps, consistent with areSteps1to4Completed
   406‚Üí   406‚Üí    const isSCompleted = (s: number): boolean => {
   407‚Üí   407‚Üí      for (let ms = 1; ms <= 5; ms++) {
   408‚Üí   408‚Üí        // Check zone-level progress first
   409‚Üí   409‚Üí        if (isStepCompletedAt(s, ms)) continue
   410‚Üí   410‚Üí        // Also check employeeProgress for this step (e.g., individual formaci√≥n step 1)
   411‚Üí   411‚Üí        const anyEmpCompleted = currentZone
   412‚Üí   412‚Üí          ? employeeProgress.some(ep =>
   413‚Üí   413‚Üí              ep.sStep === s &&
   414‚Üí   414‚Üí              ep.miniStep === ms &&
   415‚Üí   415‚Üí              ep.zoneId === currentZone.id &&
   416‚Üí   416‚Üí              ep.completed
   417‚Üí   417‚Üí            )
   418‚Üí   418‚Üí          : employeeProgress.some(ep =>
   419‚Üí   419‚Üí              ep.sStep === s &&
   420‚Üí   420‚Üí              ep.miniStep === ms &&
   421‚Üí   421‚Üí              ep.completed
   422‚Üí   422‚Üí            )
   423‚Üí   423‚Üí        if (!anyEmpCompleted) return false
   424‚Üí   424‚Üí      }
   425‚Üí   425‚Üí      return true
   426‚Üí   426‚Üí    }
   427‚Üí   427‚Üí
   428‚Üí   428‚Üí    // Check if the CURRENT USER has completed step 1 individually (EmployeeProgress)
   429‚Üí   429‚Üí    // This is crucial for unlocking step 2 for an employee who passed the exam
   430‚Üí   430‚Üí    // even if not ALL employees in the zone have completed it yet
   431‚Üí   431‚Üí    const hasUserCompletedStep1 = (): boolean => {
   432‚Üí   432‚Üí      if (!currentZone) return false
   433‚Üí   433‚Üí      const userStep = employeeProgress.find(ep =>
   434‚Üí   434‚Üí        ep.sStep === sStep &&
   435‚Üí   435‚Üí        ep.miniStep === 1 &&
   436‚Üí   436‚Üí        ep.zoneId === currentZone.id &&
   437‚Üí   437‚Üí        ep.userId === currentUser.id &&
   438‚Üí   438‚Üí        ep.completed
   439‚Üí   439‚Üí      )
   440‚Üí   440‚Üí      return !!userStep
   441‚Üí   441‚Üí    }
   442‚Üí   442‚Üí
   443‚Üí   443‚Üí    // Check if already completed at zone level (or project level if no zone)
   444‚Üí   444‚Üí    const isStepCompleted = (): boolean => isStepCompletedAt(sStep, miniStep)
   445‚Üí   445‚Üí
   446‚Üí   446‚Üí    // Check if the CURRENT USER has completed this individual step (EmployeeProgress)
   447‚Üí   447‚Üí    // For step 1 (formaci√≥n) and step 4 (autoevaluaci√≥n), completion is tracked per employee
   448‚Üí   448‚Üí    const hasUserCompletedIndividualStep = (): boolean => {
   449‚Üí   449‚Üí      if (!currentZone) return false
   450‚Üí   450‚Üí      if (miniStep !== 1 && miniStep !== 4) return false // Only individual steps
   451‚Üí   451‚Üí      const userStep = employeeProgress.find(ep =>
   452‚Üí   452‚Üí        ep.sStep === sStep &&
   453‚Üí   453‚Üí        ep.miniStep === miniStep &&
   454‚Üí   454‚Üí        ep.zoneId === currentZone.id &&
   455‚Üí   455‚Üí        ep.userId === currentUser.id &&
   456‚Üí   456‚Üí        ep.completed
   457‚Üí   457‚Üí      )
   458‚Üí   458‚Üí      return !!userStep
   459‚Üí   459‚Üí    }
   460‚Üí   460‚Üí
   461‚Üí   461‚Üí    // Combined: step is "done" if either zone-level completed OR user completed individually
   462‚Üí   462‚Üí    const isStepDoneForUser = (): boolean => isStepCompleted() || hasUserCompletedIndividualStep()
   463‚Üí   463‚Üí
   464‚Üí   464‚Üí    // Check if step is completed by ANYONE (zone or any employee)
   465‚Üí   465‚Üí    // This is crucial: ALL users must see completed steps in green,
   466‚Üí   466‚Üí    // even if they don't have permission to enter them (e.g., auditors viewing steps 1-4)
   467‚Üí   467‚Üí    const isStepCompletedByAnyone = (): boolean => {
   468‚Üí   468‚Üí      if (isStepCompleted()) return true
   469‚Üí   469‚Üí      if (!currentZone) {
   470‚Üí   470‚Üí        return employeeProgress.some(ep =>
   471‚Üí   471‚Üí          ep.sStep === sStep &&
   472‚Üí   472‚Üí          ep.miniStep === miniStep &&
   473‚Üí   473‚Üí          ep.completed
   474‚Üí   474‚Üí        )
   475‚Üí   475‚Üí      }
   476‚Üí   476‚Üí      return employeeProgress.some(ep =>
   477‚Üí   477‚Üí        ep.sStep === sStep &&
   478‚Üí   478‚Üí        ep.miniStep === miniStep &&
   479‚Üí   479‚Üí        ep.zoneId === currentZone.id &&
   480‚Üí   480‚Üí        ep.completed
   481‚Üí   481‚Üí      )
   482‚Üí   482‚Üí    }
   483‚Üí   483‚Üí
   484‚Üí   484‚Üí    // Helper: check if steps 1-4 are all completed for this S-step
   485‚Üí   485‚Üí    // Uses BOTH zone-level progress AND any employee progress,
   486‚Üí   486‚Üí    // so auditors can access step 5 when employees have completed 1-4
   487‚Üí   487‚Üí    const areSteps1to4Completed = (): boolean => {
   488‚Üí   488‚Üí      for (let ms = 1; ms <= 4; ms++) {
   489‚Üí   489‚Üí        const zoneCompleted = isStepCompletedAt(sStep, ms)
   490‚Üí   490‚Üí        if (zoneCompleted) continue
   491‚Üí   491‚Üí        // Check if ANY employee has completed this step individually
   492‚Üí   492‚Üí        const anyEmpCompleted = currentZone
   493‚Üí   493‚Üí          ? employeeProgress.some(ep =>
   494‚Üí   494‚Üí              ep.sStep === sStep &&
   495‚Üí   495‚Üí              ep.miniStep === ms &&
   496‚Üí   496‚Üí              ep.zoneId === currentZone.id &&
   497‚Üí   497‚Üí              ep.completed
   498‚Üí   498‚Üí            )
   499‚Üí   499‚Üí          : employeeProgress.some(ep =>
   500‚Üí   500‚Üí              ep.sStep === sStep &&
   501‚Üí   501‚Üí              ep.miniStep === ms &&
   502‚Üí   502‚Üí              ep.completed
   503‚Üí   503‚Üí            )
   504‚Üí   504‚Üí        if (!anyEmpCompleted) return false
   505‚Üí   505‚Üí      }
   506‚Üí   506‚Üí      return true
   507‚Üí   507‚Üí    }
   508‚Üí   508‚Üí
   509‚Üí   509‚Üí    // ‚ïê‚ïê‚ïê INTER-S PROGRESSION CHECK ‚ïê‚ïê‚ïê
   510‚Üí   510‚Üí    // S1 is always available. S2 requires S1 completed. S3 requires S2. Etc.
   511‚Üí   511‚Üí    // This enforces the 5S methodology sequence: Clasificar ‚Üí Ordenar ‚Üí Limpiar ‚Üí Estandarizar ‚Üí Mantener
   512‚Üí   512‚Üí    const isPreviousSCompleted = (): boolean => {
   513‚Üí   513‚Üí      if (sStep === 1) return true // S1 has no prerequisite
   514‚Üí   514‚Üí      return isSCompleted(sStep - 1)
   515‚Üí   515‚Üí    }
   516‚Üí   516‚Üí
   517‚Üí   517‚Üí    // ‚ïê‚ïê‚ïê INTRA-S PROGRESSION CHECK ‚ïê‚ïê‚ïê
   518‚Üí   518‚Üí    // Step 1: always available (if you have permission AND previous S is done)
   519‚Üí   519‚Üí    // Steps 2-4: need the immediately previous step completed
   520‚Üí   520‚Üí    // Step 5: needs ALL steps 1-4 completed
   521‚Üí   521‚Üí    // IMPORTANT: Completion checks include BOTH zone-level AND any employee progress,
   522‚Üí   522‚Üí    // so auditors/responsables can see steps as available when employees completed them
   523‚Üí   523‚Üí    const isPreviousStepCompleted = (): boolean => {
   524‚Üí   524‚Üí      if (miniStep === 1) return true
   525‚Üí   525‚Üí      if (miniStep === 5) return areSteps1to4Completed()
   526‚Üí   526‚Üí      // Steps 2, 3, 4: need previous step completed at zone level
   527‚Üí   527‚Üí      if (isStepCompletedAt(sStep, miniStep - 1)) return true
   528‚Üí   528‚Üí      // For step 2: current user completed step 1 individually (exam passed)
   529‚Üí   529‚Üí      if (miniStep === 2 && hasUserCompletedStep1()) return true
   530‚Üí   530‚Üí      // Also check: ANY employee completed previous step (for auditors/responsables)
   531‚Üí   531‚Üí      const anyEmpCompletedPrev = currentZone
   532‚Üí   532‚Üí        ? employeeProgress.some(ep =>
   533‚Üí   533‚Üí            ep.sStep === sStep &&
   534‚Üí   534‚Üí            ep.miniStep === miniStep - 1 &&
   535‚Üí   535‚Üí            ep.zoneId === currentZone.id &&
   536‚Üí   536‚Üí            ep.completed
   537‚Üí   537‚Üí          )
   538‚Üí   538‚Üí        : employeeProgress.some(ep =>
   539‚Üí   539‚Üí            ep.sStep === sStep &&
   540‚Üí   540‚Üí            ep.miniStep === miniStep - 1 &&
   541‚Üí   541‚Üí            ep.completed
   542‚Üí   542‚Üí          )
   543‚Üí   543‚Üí      if (anyEmpCompletedPrev) return true
   544‚Üí   544‚Üí      return false
   545‚Üí   545‚Üí    }
   546‚Üí   546‚Üí
   547‚Üí   547‚Üí    // ‚îÄ‚îÄ If already completed (zone-level OR user's individual), VERIFY COHERENCY ‚îÄ‚îÄ
   548‚Üí   548‚Üí    if (isStepDoneForUser()) {
   549‚Üí   549‚Üí      const isChainCoherent = (): boolean => {
   550‚Üí   550‚Üí        // First check inter-S: previous S must be completed
   551‚Üí   551‚Üí        if (!isPreviousSCompleted()) return false
   552‚Üí   552‚Üí        // Then check intra-S chain ‚Äî reuse same logic as isPreviousStepCompleted
   553‚Üí   553‚Üí        if (miniStep === 1) return true // Step 1 is always coherent if previous S is done
   554‚Üí   554‚Üí        if (miniStep === 5) return areSteps1to4Completed()
   555‚Üí   555‚Üí        if (isStepCompletedAt(sStep, miniStep - 1)) return true
   556‚Üí   556‚Üí        if (miniStep === 2 && hasUserCompletedStep1()) return true
   557‚Üí   557‚Üí        // Also check any employee completed previous step
   558‚Üí   558‚Üí        const anyEmpPrev = currentZone
   559‚Üí   559‚Üí          ? employeeProgress.some(ep =>
   560‚Üí   560‚Üí              ep.sStep === sStep &&
   561‚Üí   561‚Üí              ep.miniStep === miniStep - 1 &&
   562‚Üí   562‚Üí              ep.zoneId === currentZone.id &&
   563‚Üí   563‚Üí              ep.completed
   564‚Üí   564‚Üí            )
   565‚Üí   565‚Üí          : employeeProgress.some(ep =>
   566‚Üí   566‚Üí              ep.sStep === sStep &&
   567‚Üí   567‚Üí              ep.miniStep === miniStep - 1 &&
   568‚Üí   568‚Üí              ep.completed
   569‚Üí   569‚Üí            )
   570‚Üí   570‚Üí        if (anyEmpPrev) return true
   571‚Üí   571‚Üí        return false
   572‚Üí   572‚Üí      }
   573‚Üí   573‚Üí
   574‚Üí   574‚Üí      if (isChainCoherent()) {
   575‚Üí   575‚Üí        if (canPerformStep || canSkipSteps) return 'completed'
   576‚Üí   576‚Üí        if (canViewStep) return 'completed_viewonly'
   577‚Üí   577‚Üí        return 'locked'
   578‚Üí   578‚Üí      }
   579‚Üí   579‚Üí      // Chain is incoherent: treat as NOT completed ‚Äî fall through
   580‚Üí   580‚Üí    }
   581‚Üí   581‚Üí
   582‚Üí   582‚Üí    // ‚îÄ‚îÄ Step completed by ANYONE (zone or employee) ‚Üí show as green for ALL users ‚îÄ‚îÄ
   583‚Üí   583‚Üí    // Even users without permission must see completed steps in green
   584‚Üí   584‚Üí    if (isStepCompletedByAnyone()) {
   585‚Üí   585‚Üí      if (canPerformStep || canSkipSteps) return 'completed'
   586‚Üí   586‚Üí      // Everyone else sees it as completed_viewonly (green, read-only)
   587‚Üí   587‚Üí      return 'completed_viewonly'
   588‚Üí   588‚Üí    }
   589‚Üí   589‚Üí
   590‚Üí   590‚Üí    // ‚îÄ‚îÄ Admin with lock open: skip all checks ‚îÄ‚îÄ
   591‚Üí   591‚Üí    if (skipLocks) return 'available'
   592‚Üí   592‚Üí
   593‚Üí   593‚Üí    // ‚îÄ‚îÄ No permission at all = locked ‚îÄ‚îÄ
   594‚Üí   594‚Üí    if (!canViewStep && !canPerformStep) return 'locked'
   595‚Üí   595‚Üí
   596‚Üí   596‚Üí    // ‚îÄ‚îÄ Has perform permission (a1) ‚Üí check progression ‚îÄ‚îÄ
   597‚Üí   597‚Üí    if (canPerformStep) {
   598‚Üí   598‚Üí      // INTER-S: can't start this S until previous S is completed
