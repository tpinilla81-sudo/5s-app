   454→ const handleCreateCompanyFull = async () => {
   455→  if (!newCompanyName.trim()) return
   456→  setIsCreatingCompany(true)
   457→  let companyId: string | null = null
   458→  try {
   459→   // Step 1: Create company
   460→   const companyRes = await fetch('/api/companies', {
   461→    method: 'POST',
   462→    headers: { 'Content-Type': 'application/json' },
   463→    body: JSON.stringify({ name: newCompanyName, description: newCompanyDesc || undefined, nif: newCompanyNif || undefined, sector: newCompanySector || undefined, city: newCompanyCity || undefined, phone: newCompanyPhone || undefined }),
   464→   })
   465→   const companyData = await companyRes.json()
   466→   if (!companyRes.ok) {
   467→    toast.error('Error al crear empresa', { description: companyData.error || 'Error desconocido', duration: 5000 })
   468→    return
   469→   }
   470→   companyId = companyData.company?.id || companyData.id
   471→
   472→   // Step 2: Create or assign admin
   473→   let adminUserId: string | null = null
   474→   if (adminMode === 'create') {
   475→    if (!newAdminName.trim() || !newAdminEmail.trim() || !newAdminPassword.trim()) {
   476→     toast.warning('Datos incompletos', { description: 'Complete los datos del administrador', duration: 4000 })
   477→     // Company was created but admin data is missing — still close dialog and refresh
   478→     setShowNewCompany(false)
   479→     resetNewCompanyForm()
   480→     await loadStats()
   481→     await fetchCompanies()
   482→     return
   483→    }
   484→    // Create user with role 'admin' via /api/users
   485→    const userRes = await fetch('/api/users', {
   486→     method: 'POST',
   487→     headers: { 'Content-Type': 'application/json' },
   488→     body: JSON.stringify({
   489→      name: newAdminName,
   490→      email: newAdminEmail,
   491→      password: newAdminPassword,
   492→      role: 'admin',
   493→      active: true,
   494→     }),
   495→    })
   496→    const userData = await userRes.json()
   497→    if (!userRes.ok) {
   498→     toast.error('Error al crear admin', { description: userData.error || 'Error desconocido', duration: 5000 })
   499→     // Company was created but admin creation failed — still close dialog and refresh
   500→     setShowNewCompany(false)
   501→     resetNewCompanyForm()
   502→     await loadStats()
   503→     await fetchCompanies()
   504→     return
   505→    }
   506→    adminUserId = userData.user?.id
   507→   } else {
   508→    // Assign existing user
   509→    if (!assignUserId) {
   510→     toast.warning('Seleccionar usuario', { description: 'Seleccione un usuario para asignar como administrador', duration: 4000 })
   511→     setShowNewCompany(false)
   512→     resetNewCompanyForm()
   513→     await loadStats()
   514→     await fetchCompanies()
   515→     return
   516→    }
   517→    adminUserId = assignUserId
   518→    // Update user role to 'admin' if not already
   519→    const existingUser = users.find(u => u.id === assignUserId)
   520→    if (existingUser && existingUser.role !== 'admin') {
   521→     await fetch('/api/users', {
   522→      method: 'PUT',
   523→      headers: { 'Content-Type': 'application/json' },
   524→      body: JSON.stringify({ id: assignUserId, role: 'admin' }),
   525→     })
   526→    }
   527→   }
   528→
   529→   // Step 3: Add admin to company as admin_empresa
   530→   if (adminUserId && companyId) {
   531→    const memberRes = await fetch(`/api/companies/${companyId}/members`, {
   532→     method: 'POST',
   533→     headers: { 'Content-Type': 'application/json' },
   534→     body: JSON.stringify({ userId: adminUserId, role: 'admin_empresa' }),
   535→    })
   536→    if (!memberRes.ok) {
   537→     const memberData = await memberRes.json()
   538→     console.error('Error adding admin to company:', memberData.error)
   539→    }
   540→   }
   541→
   542→   // Step 4: Create subscription
   543→   if (companyId) {
   544→    const subRes = await fetch('/api/subscriptions', {
   545→     method: 'POST',
   546→     headers: { 'Content-Type': 'application/json' },
   547→     body: JSON.stringify({
   548→      companyId,
   549→      plan: newSubPlan,
   550→      status: newSubStatus,
   551→      maxUsers: newSubMaxUsers,
   552→      maxProjects: newSubMaxProjects,
   553→      price: newSubPrice,
   554→      endDate: newSubEndDate || undefined,
   555→      trialEndsAt: newSubTrialEndsAt || undefined,
   556→      notes: newSubNotes || undefined,
   557→     }),
   558→    })
   559→    if (!subRes.ok) {
   560→     console.error('Error creating subscription')
   561→    }
   562→   }
   563→
   564→   // Note: Welcome email is NOT sent automatically.
   565→   // The gestor can send it manually using the "Enviar Email" button on the company card.
   566→
   567→   setShowNewCompany(false)
   568→   resetNewCompanyForm()
