   178тЖТ   178тЖТconst normalize = (obj: PermMap): string => {
   179тЖТ   179тЖТ  const sorted: PermMap = {}
   180тЖТ   180тЖТ  Object.keys(obj).sort().forEach(role => {
   181тЖТ   181тЖТ    sorted[role] = {}
   182тЖТ   182тЖТ    Object.keys(obj[role]).sort().forEach(perm => {
   183тЖТ   183тЖТ      sorted[role][perm] = obj[role][perm]
   184тЖТ   184тЖТ    })
   185тЖТ   185тЖТ  })
   186тЖТ   186тЖТ  return JSON.stringify(sorted)
