   396→  const handleAddItem = async () => {

   410→    const isNecesario = sStep === 1 && newItem.category === 'necesario';

   412→    const qtyUnneeded = isInnecesario ? qty : (newItem.quantityUnneeded || 0);

   414→    // S1: auto-set decision to extra field (only for innecesario)

   434→          zoneId: currentZone?.id || null,
   435→          name: newItem.name,
   436→          location: newItem.location,
   437→          category: newItem.category || '',
   438→          quantity: qty,
   439→          quantityNeeded: qtyNeeded,
   440→          quantityUnneeded: qtyUnneeded,
   441→          price: newItem.price || null,
   442→          action: newItem.action || (isInnecesario ? (extra.decision || 'Jaula') : ''),

   444→          // Only Jaula items get jaula entry/quarantine; Eliminar/Tirar items go directly to Residuo

   448→          zonaOrigen: newItem.zonaOrigen || currentZone?.name || null,
   449→          zonaDestino: isInnecesario ? getZonaDestino(extra.decision as string | undefined) : (newItem.zonaOrigen || currentZone?.name || null),
   450→        }),
   451→      });
   452→
   453→      const json = await res.json();

   458→      } else {
