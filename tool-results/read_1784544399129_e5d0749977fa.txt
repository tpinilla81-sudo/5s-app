   172→  // Helper: import template items into the database so they get real IDs and become editable
   173→  const importTemplateItems = async (templateItems: any[]) => {
   174→    if (!currentProject?.id || !templateItems.length) return;
   175→    try {
   176→      // Check if items already exist in DB for this step/project/zone — avoid duplicates
   177→      const existingRes = await fetch(`/api/inventory?sStep=${sStep}&projectId=${currentProject.id}${currentZone?.id ? `&zoneId=${currentZone.id}` : ''}`);
   178→      const existingJson = await existingRes.json();
   179→      const existingNames = new Set(
   180→        (existingJson.success ? existingJson.data : []).map((i: any) => i.name?.trim().toLowerCase())
   181→      );
   182→
   183→      // Only import items that don't already exist in the DB
   184→      const itemsToCreate = templateItems.filter(
   185→        (item: any) => item.name && !existingNames.has(item.name.trim().toLowerCase())
   186→      );
   187→
   188→      if (itemsToCreate.length === 0) {
   189→        // All items already in DB — just reload from API to get proper IDs
   190→        await loadInventory();
   191→        return;
   192→      }
   193→
   194→      // Create each item via POST so they get database IDs
   195→      for (const item of itemsToCreate) {
   196→        const isInnecesario = sStep === 1 && item.category === 'innecesario';
   197→        const isNecesario = sStep === 1 && item.category === 'necesario';
   198→        const qty = item.quantity || 1;
   199→        const extra = { ...(item.extra || {}) };
   200→        if (sStep === 1 && isInnecesario && !extra.decision) {
   201→          extra.decision = 'Jaula';
   202→        }
   203→
   204→        await fetch('/api/inventory', {
   205→          method: 'POST',
   206→          headers: { 'Content-Type': 'application/json' },
   207→          body: JSON.stringify({
   208→            sStep,
   209→            projectId: currentProject.id,
   210→            zoneId: currentZone?.id || null,
   211→            name: item.name,
   212→            location: item.location || '',
   213→            category: item.category || '',
   214→            quantity: qty,
   215→            quantityNeeded: isNecesario ? qty : (item.quantityNeeded || 0),
   216→            quantityUnneeded: isInnecesario ? qty : (item.quantityUnneeded || 0),
   217→            price: item.price ?? null,
   218→            action: item.action || (isInnecesario ? (extra.decision || 'Jaula') : ''),
   219→            extra,
   220→            jaulaStatus: isInnecesario && extra.decision !== 'Eliminar' && extra.decision !== 'Tirar' ? 'en_jaula' : '',
   221→            jaulaFechaEntrada: isInnecesario && extra.decision !== 'Eliminar' && extra.decision !== 'Tirar' ? new Date().toISOString() : null,
   222→            jaulaOrigen: isInnecesario ? (currentZone?.name || currentProject.name || '') : null,
   223→            zonaOrigen: currentZone?.name || null,
   224→            zonaDestino: isInnecesario ? (extra.decision === 'Tirar' || extra.decision === 'Eliminar' ? 'Residuo' : 'Jaula') : null,
   225→          }),
   226→        });
   227→      }
   228→
   229→      // Now reload from API — items will have real database IDs and be fully editable
   230→      await loadInventory();
   231→    } catch (e) {
   232→      console.error('Error importing template items:', e);
   233→      // Fallback: just reload from API
   234→      await loadInventory();
   235→    }

   243→        const slotsJson = await slotsRes.json();

   245→          const slot = slotsJson.data[0];
   246→          const inventarioTemplates = (slot.templates || []).filter(
   247→            (t: any) => t.template?.type === 'inventario'

   254→                subtitle: content.subtitle || defaultConfig.subtitle,

   259→              });

   261→              // Auto-import template items into DB so they get IDs and become editable

   263→                await importTemplateItems(content.items);
   264→              }

   266→              // Legacy format: {items: [...]} — use default config but load items
   267→              setCustomConfig(null);
   268→              setHasTemplate(true);
   269→              if (content.items.length > 0) {
   270→                await importTemplateItems(content.items);
   271→              }
   272→            } else {
   273→              // Unknown format — use default config as fallback
   274→              setCustomConfig(null);
   275→              setHasTemplate(true);
   276→            }

   278→            // No inventario template assigned in this board slot — use default config
   279→            setCustomConfig(null);
   280→            setHasTemplate(true);
   281→          }
   282→        } else {

   292→          const content = JSON.parse(json.data[0].content);
   293→          if (content.categories && content.extraFields) {
   294→            setCustomConfig({
   295→              title: content.title || defaultConfig.title,
   296→              subtitle: content.subtitle || defaultConfig.subtitle,
   297→              templateName: content.templateName || defaultConfig.templateName,
   298→              categories: content.categories,
   299→              extraFields: content.extraFields,
   300→              ...(content.desplegables_jerarquicos ? { desplegables_jerarquicos: content.desplegables_jerarquicos } : {}),
   301→            });
   302→            setHasTemplate(true);
   303→            // Auto-import template items into DB so they get IDs and become editable
   304→            if (content.items && Array.isArray(content.items) && content.items.length > 0) {
   305→              await importTemplateItems(content.items);
   306→            }
   307→          } else if (content.items && Array.isArray(content.items)) {
   308→            // Legacy format: {items: [...]} — use default config but load items
   309→            setCustomConfig(null);
   310→            setHasTemplate(true);
   311→            if (content.items.length > 0) {
   312→              await importTemplateItems(content.items);
   313→            }
   314→          } else {
   315→            // Unknown format — use default config as fallback
   316→            setCustomConfig(null);
   317→            setHasTemplate(true);
   318→          }

   320→          // No global template — use default config (INVENTORY_CONFIGS has entries for all 5 S steps)

   323→        }
   324→      }
   325→    } catch (e) {

   423→  const loadInventory = async () => {

   430→      if (json.success) {

   432→        const mappedItems = json.data.map((item: any) => {
   433→          // Map photos from the relation
   434→          const itemPhotosList: PhotoData[] = (item.photos || []).map((p: any) => ({

   436→            title: p.title,
   437→            description: p.description,
   438→            photoUrl: p.photoUrl,

   440→            category: p.category,
   441→            inventoryItemId: p.inventoryItemId,
   442→            miniStep: p.miniStep,
   443→            sStep: p.sStep,
   444→            createdAt: p.createdAt,
   445→          }));
   446→          if (item.id && itemPhotosList.length > 0) {
   447→            photosMap[item.id] = itemPhotosList;
   448→          }
   449→          return {
   450→            id: item.id,
   451→            name: item.name,
   452→            location: item.location || '',
   453→            category: item.category || '',
   454→            quantity: item.quantity || 1,
   455→            // S1: Set quantities based on category (innecesario or necesario)
   456→            quantityNeeded: sStep === 1
   457→              ? (item.category === 'necesario' ? (item.quantityNeeded || item.quantity || 1) : 0)
   458→              : (item.quantityNeeded || 0),
   459→            quantityUnneeded: sStep === 1
   460→              ? (item.category === 'innecesario' ? (item.quantityUnneeded || item.quantity || 1) : 0)
   461→              : (item.quantityUnneeded || 0),
   462→            price: item.price ?? null,
   463→            action: item.action || '',
   464→            extra: typeof item.extra === 'string' ? JSON.parse(item.extra) : (item.extra || {}),
   465→            jaulaStatus: item.jaulaStatus || '',
   466→            jaulaFechaEntrada: item.jaulaFechaEntrada || null,
   467→            jaulaOrigen: item.jaulaOrigen || null,
   468→            jaulaFechaSalida: item.jaulaFechaSalida || null,
   469→            jaulaDestino: item.jaulaDestino || null,
   470→            zonaOrigen: item.zonaOrigen || null,

   472→            photos: itemPhotosList,
   473→          };
   474→        });
   475→        setItems(mappedItems);
   476→        setItemPhotos(photosMap);
   477→      } else {
   478→        console.error('Error loading inventory:', json.error);
   479→      }
   480→    } catch (error) {
   481→      console.error('Error loading inventory:', error);
   482→    } finally {
   483→      setIsLoading(false);
   484→    }

   891→            return v !== '' && v !== '0';

   894→          // This filters out empty numbered rows and footer rows

   940→        fuente: findCol(headerRow, 'fuente'),

   944→        ubicacionAsignada: findCol(headerRow, 'ubicación asignada', 'asignada'),
