   358→// VISUAL EDITOR: ChecklistEditor (autoevaluacion / auditoria)

   360→interface ChecklistSection {
   361→  id: string
   362→  title: string
   363→  items: { id: string; description: string; hasOther: boolean }[]
   364→}
   365→
   366→function ChecklistEditor({ content, onChange }: { content: string; onChange: (v: string) => void }) {
   367→  let parsed: { sections: ChecklistSection[] }
   368→  try {
   369→    parsed = JSON.parse(content)
   370→  } catch {
   371→    return (
   372→      <div className="p-4 bg-red-50 border border-red-200 rounded-lg text-sm text-red-600">
   373→        El JSON no es válido. Corrígelo en modo JSON o carga el contenido por defecto.
   374→      </div>
   375→    )
   376→  }
   377→
   378→  const sections = parsed.sections || []
   379→  const sPrefix = String(content.match(/"id"\s*:\s*"(\d+)\./)?.[1] || '1')
   380→
   381→  const update = (newSections: ChecklistSection[]) => {
   382→    onChange(JSON.stringify({ ...parsed, sections: newSections }, null, 2))
   383→  }
   384→
   385→  const addSection = () => {
   386→    const newId = `${sPrefix}.${sections.length + 1}`
   387→    update([...sections, { id: newId, title: 'Nueva Sección', items: [] }])
   388→  }
   389→
   390→  const removeSection = (idx: number) => {
   391→    update(sections.filter((_, i) => i !== idx))
   392→  }
   393→
   394→  const updateSection = (idx: number, field: 'id' | 'title', value: string) => {
   395→    const updated = [...sections]
   396→    updated[idx] = { ...updated[idx], [field]: value }
   397→    update(updated)
   398→  }
   399→
   400→  const moveSection = (idx: number, dir: -1 | 1) => {
   401→    const target = idx + dir
   402→    if (target < 0 || target >= sections.length) return
   403→    const updated = [...sections]
   404→    ;[updated[idx], updated[target]] = [updated[target], updated[idx]]
   405→    update(updated)
   406→  }
   407→
   408→  const addItem = (sectionIdx: number) => {
   409→    const sec = sections[sectionIdx]
   410→    const newItemId = `${sec.id}.${sec.items.length + 1}`
   411→    const updated = [...sections]
   412→    updated[sectionIdx] = {
   413→      ...updated[sectionIdx],
   414→      items: [...updated[sectionIdx].items, { id: newItemId, description: '', hasOther: false }]
   415→    }
   416→    update(updated)
   417→  }
   418→
   419→  const removeItem = (sectionIdx: number, itemIdx: number) => {
   420→    const updated = [...sections]
   421→    updated[sectionIdx] = {
   422→      ...updated[sectionIdx],
   423→      items: updated[sectionIdx].items.filter((_, i) => i !== itemIdx)
   424→    }
   425→    update(updated)
   426→  }
   427→
   428→  const updateItem = (sectionIdx: number, itemIdx: number, field: string, value: string | boolean) => {
   429→    const updated = [...sections]
   430→    updated[sectionIdx] = {
   431→      ...updated[sectionIdx],
   432→      items: updated[sectionIdx].items.map((item, i) =>
   433→        i === itemIdx ? { ...item, [field]: value } : item
   434→      )
   435→    }
   436→    update(updated)
   437→  }
   438→
   439→  const moveItem = (sectionIdx: number, itemIdx: number, dir: -1 | 1) => {
   440→    const items = sections[sectionIdx].items
   441→    const target = itemIdx + dir
   442→    if (target < 0 || target >= items.length) return
   443→    const updated = [...sections]
   444→    const newItems = [...updated[sectionIdx].items]
   445→    ;[newItems[itemIdx], newItems[target]] = [newItems[target], newItems[itemIdx]]
   446→    updated[sectionIdx] = { ...updated[sectionIdx], items: newItems }
   447→    update(updated)
   448→  }
   449→
   450→  return (
   451→    <div className="space-y-4">
   452→      {sections.map((section, sIdx) => (
   453→        <div key={sIdx} className="border-2 rounded-lg overflow-hidden bg-white shadow-sm">
   454→          {/* Section header */}
   455→          <div className="flex items-center gap-3 px-4 py-3 bg-gray-50 border-b">
   456→            <div className="flex flex-col gap-0.5">
   457→              <button onClick={() => moveSection(sIdx, -1)} className="text-gray-400 hover:text-gray-600 leading-none" title="Subir sección">
   458→                <ChevronUp className="h-4 w-4" />
   459→              </button>
   460→              <button onClick={() => moveSection(sIdx, 1)} className="text-gray-400 hover:text-gray-600 leading-none" title="Bajar sección">
   461→                <ChevronDown className="h-4 w-4" />
   462→              </button>
   463→            </div>
   464→            <Input
   465→              value={section.id}
   466→              onChange={(e) => updateSection(sIdx, 'id', e.target.value)}
   467→              className="w-20 h-9 text-sm font-mono"
   468→              placeholder="ID"
   469→            />
   470→            <Input
   471→              value={section.title}
   472→              onChange={(e) => updateSection(sIdx, 'title', e.target.value)}
   473→              className="flex-1 h-9 text-base font-semibold"
   474→              placeholder="Título de sección"
   475→            />
   476→            <Button variant="ghost" size="sm" onClick={() => addItem(sIdx)}
   477→              className="h-9 w-9 p-0 text-green-600 hover:text-green-700 hover:bg-green-50" title="Añadir item">
   478→              <Plus className="h-5 w-5" />
   479→            </Button>
   480→            <Button variant="ghost" size="sm" onClick={() => removeSection(sIdx)}
   481→              className="h-9 w-9 p-0 text-red-500 hover:text-red-600 hover:bg-red-50" title="Eliminar sección">
   482→              <Trash2 className="h-4 w-4" />
   483→            </Button>
   484→          </div>
   485→
   486→          {/* Items */}
   487→          <div className="p-3 space-y-2">
   488→            {section.items.length === 0 && (
   489→              <p className="text-sm text-muted-foreground italic px-2 py-2">Sin items. Pulsa + para añadir.</p>
   490→            )}
   491→            {section.items.map((item, iIdx) => (
   492→              <div key={iIdx} className="flex items-center gap-3 group">
   493→                <div className="flex flex-col gap-0.5">

   498→                    <ChevronDown className="h-3.5 w-3.5" />
   499→                  </button>

   518→                    className="rounded border-gray-300 h-4 w-4"
   519→                  />
   520→                  Otros
   521→                </label>
   522→                <Button variant="ghost" size="sm"
   523→                  onClick={() => removeItem(sIdx, iIdx)}
   524→                  className="h-8 w-8 p-0 text-red-400 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity" title="Eliminar">

   527→              </div>
   528→            ))}
   529→          </div>
   530→        </div>
   531→      ))}
   532→
   533→      <Button variant="outline" onClick={addSection}
   534→        className="w-full border-dashed border-2 text-green-600 hover:bg-green-50 hover:border-green-400 gap-1 h-10">
   535→        <Plus className="h-5 w-5" />

   540→        {sections.length} sección(es) · {sections.reduce((s, sec) => s + sec.items.length, 0)} item(s) en total
