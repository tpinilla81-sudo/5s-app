   367тЖТ   367тЖТ  //   - INTER-S progression: S2 requires S1 completed, S3 requires S2, etc.
   368тЖТ   368тЖТ  //   - INTRA-S progression: step N requires step N-1 completed

   372тЖТ   372тЖТ  //   - No permission тЖТ locked (not even visible)
   373тЖТ   373тЖТ  //   - IMPORTANT: Previous step completion is checked at ZONE level,
   374тЖТ   374тЖТ  //     so an auditor can access step 5 if employees completed 1-4
   375тЖТ   375тЖТ  //   - For step 1 (individual): also checks EmployeeProgress for the
   376тЖТ   376тЖТ  //     current user, so the employee's own completion unlocks step 2

   378тЖТ   378тЖТ  getMiniStepStatus: (sStep, miniStep) => {
   379тЖТ   379тЖТ    const { progress, employeeProgress, currentUser, adminFreeNavigation, currentZone } = get()
   380тЖТ   380тЖТ    if (!currentUser) return 'locked'
   381тЖТ   381тЖТ
   382тЖТ   382тЖТ    const canViewStep = get().canView(sStep, miniStep)
   383тЖТ   383тЖТ    const canPerformStep = get().canPerform(sStep, miniStep)
   384тЖТ   384тЖТ    // Permission-driven: skip_steps from DB replaces hardcoded isAdmin check
   385тЖТ   385тЖТ    const canSkipSteps = get().hasPermission('skip_steps')
   386тЖТ   386тЖТ    const skipLocks = canSkipSteps && adminFreeNavigation
   387тЖТ   387тЖТ
   388тЖТ   388тЖТ    // Check if a specific mini-step of a specific S is completed at zone level
   389тЖТ   389тЖТ    const isStepCompletedAt = (s: number, ms: number): boolean => {
   390тЖТ   390тЖТ      if (currentZone) {
   391тЖТ   391тЖТ        const zoneStep = progress.find(p =>
   392тЖТ   392тЖТ          p.sStep === s &&
   393тЖТ   393тЖТ          p.miniStep === ms &&
   394тЖТ   394тЖТ          (p.zoneId === currentZone.id || p.zoneId === null) &&
   395тЖТ   395тЖТ          p.completed
   396тЖТ   396тЖТ        )
   397тЖТ   397тЖТ        return !!zoneStep
   398тЖТ   398тЖТ      }
   399тЖТ   399тЖТ      const anyStep = progress.find(p =>
   400тЖТ   400тЖТ        p.sStep === s &&
   401тЖТ   401тЖТ        p.miniStep === ms &&
   402тЖТ   402тЖТ        p.completed
   403тЖТ   403тЖТ      )
   404тЖТ   404тЖТ      return !!anyStep
   405тЖТ   405тЖТ    }
   406тЖТ   406тЖТ
   407тЖТ   407тЖТ    // Check if a specific S-step (all 5 mini-steps) is fully completed (quesito earned)
   408тЖТ   408тЖТ    // Also checks employeeProgress for individual steps, consistent with areSteps1to4Completed
   409тЖТ   409тЖТ    const isSCompleted = (s: number): boolean => {
   410тЖТ   410тЖТ      for (let ms = 1; ms <= 5; ms++) {
   411тЖТ   411тЖТ        // Check zone-level progress first
   412тЖТ   412тЖТ        if (isStepCompletedAt(s, ms)) continue
   413тЖТ   413тЖТ        // Also check employeeProgress for this step (e.g., individual formaci├│n step 1)
   414тЖТ   414тЖТ        const anyEmpCompleted = currentZone
   415тЖТ   415тЖТ          ? employeeProgress.some(ep =>
   416тЖТ   416тЖТ              ep.sStep === s &&
   417тЖТ   417тЖТ              ep.miniStep === ms &&
   418тЖТ   418тЖТ              ep.zoneId === currentZone.id &&
   419тЖТ   419тЖТ              ep.completed
   420тЖТ   420тЖТ            )
   421тЖТ   421тЖТ          : employeeProgress.some(ep =>
   422тЖТ   422тЖТ              ep.sStep === s &&
   423тЖТ   423тЖТ              ep.miniStep === ms &&
   424тЖТ   424тЖТ              ep.completed
   425тЖТ   425тЖТ            )
   426тЖТ   426тЖТ        if (!anyEmpCompleted) return false
   427тЖТ   427тЖТ      }
   428тЖТ   428тЖТ      return true
   429тЖТ   429тЖТ    }
   430тЖТ   430тЖТ
   431тЖТ   431тЖТ    // Check if the CURRENT USER has completed step 1 individually (EmployeeProgress)
   432тЖТ   432тЖТ    // This is crucial for unlocking step 2 for an employee who passed the exam
   433тЖТ   433тЖТ    // even if not ALL employees in the zone have completed it yet
   434тЖТ   434тЖТ    const hasUserCompletedStep1 = (): boolean => {
   435тЖТ   435тЖТ      if (!currentZone) return false
   436тЖТ   436тЖТ      const userStep = employeeProgress.find(ep =>
   437тЖТ   437тЖТ        ep.sStep === sStep &&
   438тЖТ   438тЖТ        ep.miniStep === 1 &&
   439тЖТ   439тЖТ        ep.zoneId === currentZone.id &&
   440тЖТ   440тЖТ        ep.userId === currentUser.id &&
   441тЖТ   441тЖТ        ep.completed
   442тЖТ   442тЖТ      )
   443тЖТ   443тЖТ      return !!userStep
   444тЖТ   444тЖТ    }
   445тЖТ   445тЖТ
   446тЖТ   446тЖТ    // Check if already completed at zone level (or project level if no zone)
   447тЖТ   447тЖТ    const isStepCompleted = (): boolean => isStepCompletedAt(sStep, miniStep)
   448тЖТ   448тЖТ
   449тЖТ   449тЖТ    // Check if the CURRENT USER has completed this individual step (EmployeeProgress)
   450тЖТ   450тЖТ    // For step 1 (formaci├│n) and step 4 (autoevaluaci├│n), completion is tracked per employee
   451тЖТ   451тЖТ    const hasUserCompletedIndividualStep = (): boolean => {
   452тЖТ   452тЖТ      if (!currentZone) return false
   453тЖТ   453тЖТ      if (miniStep !== 1 && miniStep !== 4) return false // Only individual steps
   454тЖТ   454тЖТ      const userStep = employeeProgress.find(ep =>
   455тЖТ   455тЖТ        ep.sStep === sStep &&
   456тЖТ   456тЖТ        ep.miniStep === miniStep &&
   457тЖТ   457тЖТ        ep.zoneId === currentZone.id &&

   462тЖТ   462тЖТ    }
   463тЖТ   463тЖТ
   464тЖТ   464тЖТ    // Combined: step is "done" if either zone-level completed OR user completed individually
   465тЖТ   465тЖТ    const isStepDoneForUser = (): boolean => isStepCompleted() || hasUserCompletedIndividualStep()
   466тЖТ   466тЖТ
   467тЖТ   467тЖТ    // Check if step is completed by ANYONE (zone or any employee)
   468тЖТ   468тЖТ    // This is crucial: ALL users must see completed steps in green,
   469тЖТ   469тЖТ    // even if they don't have permission to enter them (e.g., auditors viewing steps 1-4)
   470тЖТ   470тЖТ    const isStepCompletedByAnyone = (): boolean => {
   471тЖТ   471тЖТ      if (isStepCompleted()) return true
   472тЖТ   472тЖТ      if (!currentZone) {
   473тЖТ   473тЖТ        return employeeProgress.some(ep =>
   474тЖТ   474тЖТ          ep.sStep === sStep &&
   475тЖТ   475тЖТ          ep.miniStep === miniStep &&
   476тЖТ   476тЖТ          ep.completed
   477тЖТ   477тЖТ        )
   478тЖТ   478тЖТ      }
   479тЖТ   479тЖТ      return employeeProgress.some(ep =>
   480тЖТ   480тЖТ        ep.sStep === sStep &&
   481тЖТ   481тЖТ        ep.miniStep === miniStep &&
   482тЖТ   482тЖТ        ep.zoneId === currentZone.id &&
   483тЖТ   483тЖТ        ep.completed

   485тЖТ   485тЖТ    }
   486тЖТ   486тЖТ
   487тЖТ   487тЖТ    // Helper: check if steps 1-4 are all completed for this S-step
   488тЖТ   488тЖТ    // Uses BOTH zone-level progress AND any employee progress,
   489тЖТ   489тЖТ    // so auditors can access step 5 when employees have completed 1-4
   490тЖТ   490тЖТ    const areSteps1to4Completed = (): boolean => {
   491тЖТ   491тЖТ      for (let ms = 1; ms <= 4; ms++) {
   492тЖТ   492тЖТ        const zoneCompleted = isStepCompletedAt(sStep, ms)
   493тЖТ   493тЖТ        if (zoneCompleted) continue
   494тЖТ   494тЖТ        // Check if ANY employee has completed this step individually
   495тЖТ   495тЖТ        const anyEmpCompleted = currentZone
   496тЖТ   496тЖТ          ? employeeProgress.some(ep =>
   497тЖТ   497тЖТ              ep.sStep === sStep &&
   498тЖТ   498тЖТ              ep.miniStep === ms &&
   499тЖТ   499тЖТ              ep.zoneId === currentZone.id &&
   500тЖТ   500тЖТ              ep.completed
   501тЖТ   501тЖТ            )
   502тЖТ   502тЖТ          : employeeProgress.some(ep =>
   503тЖТ   503тЖТ              ep.sStep === sStep &&
   504тЖТ   504тЖТ              ep.miniStep === ms &&
   505тЖТ   505тЖТ              ep.completed
   506тЖТ   506тЖТ            )
   507тЖТ   507тЖТ        if (!anyEmpCompleted) return false
   508тЖТ   508тЖТ      }
   509тЖТ   509тЖТ      return true
   510тЖТ   510тЖТ    }
   511тЖТ   511тЖТ
   512тЖТ   512тЖТ    // тХРтХРтХР INTER-S PROGRESSION CHECK тХРтХРтХР
   513тЖТ   513тЖТ    // S1 is always available. S2 requires S1 completed. S3 requires S2. Etc.
   514тЖТ   514тЖТ    // This enforces the 5S methodology sequence: Clasificar тЖТ Ordenar тЖТ Limpiar тЖТ Estandarizar тЖТ Mantener
   515тЖТ   515тЖТ    const isPreviousSCompleted = (): boolean => {
   516тЖТ   516тЖТ      if (sStep === 1) return true // S1 has no prerequisite
   517тЖТ   517тЖТ      return isSCompleted(sStep - 1)
   518тЖТ   518тЖТ    }
   519тЖТ   519тЖТ
   520тЖТ   520тЖТ    // тХРтХРтХР INTRA-S PROGRESSION CHECK тХРтХРтХР
   521тЖТ   521тЖТ    // Step 1: always available (if you have permission AND previous S is done)
   522тЖТ   522тЖТ    // Steps 2-4: need the immediately previous step completed
   523тЖТ   523тЖТ    // Step 5: needs ALL steps 1-4 completed
   524тЖТ   524тЖТ    // IMPORTANT: Completion checks include BOTH zone-level AND any employee progress,
   525тЖТ   525тЖТ    // so auditors/responsables can see steps as available when employees completed them
   526тЖТ   526тЖТ    const isPreviousStepCompleted = (): boolean => {
   527тЖТ   527тЖТ      if (miniStep === 1) return true
   528тЖТ   528тЖТ      if (miniStep === 5) return areSteps1to4Completed()
   529тЖТ   529тЖТ      // Steps 2, 3, 4: need previous step completed at zone level
   530тЖТ   530тЖТ      if (isStepCompletedAt(sStep, miniStep - 1)) return true
   531тЖТ   531тЖТ      // For step 2: current user completed step 1 individually (exam passed)
   532тЖТ   532тЖТ      if (miniStep === 2 && hasUserCompletedStep1()) return true
   533тЖТ   533тЖТ      // Also check: ANY employee completed previous step (for auditors/responsables)
   534тЖТ   534тЖТ      const anyEmpCompletedPrev = currentZone
   535тЖТ   535тЖТ        ? employeeProgress.some(ep =>
   536тЖТ   536тЖТ            ep.sStep === sStep &&
   537тЖТ   537тЖТ            ep.miniStep === miniStep - 1 &&
   538тЖТ   538тЖТ            ep.zoneId === currentZone.id &&
   539тЖТ   539тЖТ            ep.completed
   540тЖТ   540тЖТ          )
   541тЖТ   541тЖТ        : employeeProgress.some(ep =>
   542тЖТ   542тЖТ            ep.sStep === sStep &&
   543тЖТ   543тЖТ            ep.miniStep === miniStep - 1 &&
   544тЖТ   544тЖТ            ep.completed
   545тЖТ   545тЖТ          )
   546тЖТ   546тЖТ      if (anyEmpCompletedPrev) return true
   547тЖТ   547тЖТ      return false
   548тЖТ   548тЖТ    }
   549тЖТ   549тЖТ
   550тЖТ   550тЖТ    // тФАтФА If already completed (zone-level OR user's individual), VERIFY COHERENCY тФАтФА
   551тЖТ   551тЖТ    if (isStepDoneForUser()) {
   552тЖТ   552тЖТ      const isChainCoherent = (): boolean => {
   553тЖТ   553тЖТ        // First check inter-S: previous S must be completed
   554тЖТ   554тЖТ        if (!isPreviousSCompleted()) return false
   555тЖТ   555тЖТ        // Then check intra-S chain тАФ reuse same logic as isPreviousStepCompleted
   556тЖТ   556тЖТ        if (miniStep === 1) return true // Step 1 is always coherent if previous S is done
   557тЖТ   557тЖТ        if (miniStep === 5) return areSteps1to4Completed()
   558тЖТ   558тЖТ        if (isStepCompletedAt(sStep, miniStep - 1)) return true
   559тЖТ   559тЖТ        if (miniStep === 2 && hasUserCompletedStep1()) return true
   560тЖТ   560тЖТ        // Also check any employee completed previous step
   561тЖТ   561тЖТ        const anyEmpPrev = currentZone
   562тЖТ   562тЖТ          ? employeeProgress.some(ep =>
   563тЖТ   563тЖТ              ep.sStep === sStep &&
   564тЖТ   564тЖТ              ep.miniStep === miniStep - 1 &&
   565тЖТ   565тЖТ              ep.zoneId === currentZone.id &&
   566тЖТ   566тЖТ              ep.completed
   567тЖТ   567тЖТ            )
   568тЖТ   568тЖТ          : employeeProgress.some(ep =>
   569тЖТ   569тЖТ              ep.sStep === sStep &&
   570тЖТ   570тЖТ              ep.miniStep === miniStep - 1 &&
   571тЖТ   571тЖТ              ep.completed
   572тЖТ   572тЖТ            )
   573тЖТ   573тЖТ        if (anyEmpPrev) return true
   574тЖТ   574тЖТ        return false
   575тЖТ   575тЖТ      }
   576тЖТ   576тЖТ
   577тЖТ   577тЖТ      if (isChainCoherent()) {
   578тЖТ   578тЖТ        if (canPerformStep || canSkipSteps) return 'completed'
   579тЖТ   579тЖТ        if (canViewStep) return 'completed_viewonly'
   580тЖТ   580тЖТ        return 'locked'
   581тЖТ   581тЖТ      }
   582тЖТ   582тЖТ      // Chain is incoherent: treat as NOT completed тАФ fall through
   583тЖТ   583тЖТ    }
   584тЖТ   584тЖТ
   585тЖТ   585тЖТ    // тФАтФА Step completed by ANYONE (zone or employee) тЖТ show as green for ALL users тФАтФА
   586тЖТ   586тЖТ    // Even users without permission must see completed steps in green
   587тЖТ   587тЖТ    if (isStepCompletedByAnyone()) {
   588тЖТ   588тЖТ      if (canPerformStep || canSkipSteps) return 'completed'
   589тЖТ   589тЖТ      // Everyone else sees it as completed_viewonly (green, read-only)
   590тЖТ   590тЖТ      return 'completed_viewonly'
   591тЖТ   591тЖТ    }
   592тЖТ   592тЖТ
   593тЖТ   593тЖТ    // тФАтФА Admin with lock open: skip all checks тФАтФА
   594тЖТ   594тЖТ    if (skipLocks) return 'available'
   595тЖТ   595тЖТ
   596тЖТ   596тЖТ    // тФАтФА No permission at all = locked тФАтФА
   597тЖТ   597тЖТ    if (!canViewStep && !canPerformStep) return 'locked'
   598тЖТ   598тЖТ
   599тЖТ   599тЖТ    // тФАтФА Has perform permission (a1) тЖТ check progression тФАтФА
   600тЖТ   600тЖТ    if (canPerformStep) {
   601тЖТ   601тЖТ      // INTER-S: can't start this S until previous S is completed
   602тЖТ   602тЖТ      if (!isPreviousSCompleted()) return 'locked'
   603тЖТ   603тЖТ      // INTRA-S: can't enter step until previous step is completed
   604тЖТ   604тЖТ      if (!isPreviousStepCompleted()) return 'locked'
   605тЖТ   605тЖТ      return 'available'
   606тЖТ   606тЖТ    }
   607тЖТ   607тЖТ
   608тЖТ   608тЖТ    // тФАтФА Has view permission only (a0) тЖТ can see but CANNOT enter тФАтФА
   609тЖТ   609тЖТ    if (canViewStep) return 'locked'
   610тЖТ   610тЖТ
   611тЖТ   611тЖТ    return 'locked'
   612тЖТ   612тЖТ  },

   614тЖТ   614тЖТ  isZoneMiniStepComplete: (sStep, miniStep, zoneId) => {
   615тЖТ   615тЖТ    const { progress, employeeProgress, currentProject } = get()
   616тЖТ   616тЖТ    if (!currentProject) return false
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ    if (ZONE_MINI_STEPS.includes(miniStep)) {

   620тЖТ   620тЖТ      const zoneStep = progress.find(p =>

   626тЖТ   626тЖТ    }

   628тЖТ   628тЖТ    if (INDIVIDUAL_MINI_STEPS.includes(miniStep)) {
   629тЖТ   629тЖТ      // Individual step (1 only): ALL employees in the zone must have completed
   630тЖТ   630тЖТ      // The zone-level Progress record for step 1 is only marked completed when ALL employees pass
   631тЖТ   631тЖТ      const zoneStep = progress.find(p =>
   632тЖТ   632тЖТ        p.sStep === sStep &&
   633тЖТ   633тЖТ        p.miniStep === miniStep &&
   634тЖТ   634тЖТ        (p.zoneId === zoneId || p.zoneId === null) &&
   635тЖТ   635тЖТ        p.completed
   636тЖТ   636тЖТ      )
   637тЖТ   637тЖТ      return !!zoneStep
   638тЖТ   638тЖТ    }
   639тЖТ   639тЖТ
   640тЖТ   640тЖТ    return false

   643тЖТ   643тЖТ  isQuesitoEarned: (sStep) => {
   644тЖТ   644тЖТ    const { progress, employeeProgress, currentZone } = get()
   645тЖТ   645тЖТ    const zoneId = currentZone?.id || null
   646тЖТ   646тЖТ
   647тЖТ   647тЖТ    // Check ALL 5 mini-steps are completed тАФ each one individually
