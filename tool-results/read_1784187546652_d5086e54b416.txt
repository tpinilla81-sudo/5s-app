   350→  canPerform: (sStep: number, miniStep: number): boolean => {

   360→  // ═══════════════════════════════════════════════════════
   361→  // getMiniStepStatus — Permission-Driven with Business Rules
   362→  // Permissions are the source of truth for WHO can do WHAT.
   363→  // Business rules define WHEN things are accessible:
   364→  //   - INTER-S progression: S2 requires S1 completed, S3 requires S2, etc.
   365→  //   - INTRA-S progression: step N requires step N-1 completed
   366→  //   - Step 5 (audit): requires steps 1-4 ALL completed within the same S
   367→  //   - a1 (perform) → can enter and act (if previous step done)
   368→  //   - a0 (view) only → can see the step exists, but CANNOT enter it
   369→  //   - No permission → locked (not even visible)
   370→  //   - IMPORTANT: Previous step completion is checked at ZONE level,
   371→  //     so an auditor can access step 5 if employees completed 1-4
   372→  //   - For step 1 (individual): also checks EmployeeProgress for the
   373→  //     current user, so the employee's own completion unlocks step 2
   374→  // ═══════════════════════════════════════════════════════
   375→  getMiniStepStatus: (sStep, miniStep) => {
   376→    const { progress, employeeProgress, currentUser, adminFreeNavigation, currentZone } = get()
   377→    if (!currentUser) return 'locked'
   378→
   379→    const canViewStep = get().canView(sStep, miniStep)
   380→    const canPerformStep = get().canPerform(sStep, miniStep)
   381→    // Permission-driven: skip_steps from DB replaces hardcoded isAdmin check
   382→    const canSkipSteps = get().hasPermission('skip_steps')
   383→    const skipLocks = canSkipSteps && adminFreeNavigation
   384→
   385→    // Check if a specific mini-step of a specific S is completed at zone level
   386→    const isStepCompletedAt = (s: number, ms: number): boolean => {
   387→      if (currentZone) {
   388→        const zoneStep = progress.find(p =>
   389→          p.sStep === s &&
   390→          p.miniStep === ms &&
   391→          (p.zoneId === currentZone.id || p.zoneId === null) &&
   392→          p.completed
   393→        )
   394→        return !!zoneStep
   395→      }
   396→      const anyStep = progress.find(p =>
   397→        p.sStep === s &&
   398→        p.miniStep === ms &&
   399→        p.completed
   400→      )
   401→      return !!anyStep
   402→    }
   403→
   404→    // Check if a specific S-step (all 5 mini-steps) is fully completed (quesito earned)
   405→    // Also checks employeeProgress for individual steps, consistent with areSteps1to4Completed
   406→    const isSCompleted = (s: number): boolean => {
   407→      for (let ms = 1; ms <= 5; ms++) {
   408→        // Check zone-level progress first
   409→        if (isStepCompletedAt(s, ms)) continue
   410→        // Also check employeeProgress for this step (e.g., individual formación step 1)
   411→        const anyEmpCompleted = currentZone
   412→          ? employeeProgress.some(ep =>
   413→              ep.sStep === s &&
   414→              ep.miniStep === ms &&
   415→              ep.zoneId === currentZone.id &&
   416→              ep.completed
   417→            )
   418→          : employeeProgress.some(ep =>
   419→              ep.sStep === s &&
   420→              ep.miniStep === ms &&
   421→              ep.completed
   422→            )
   423→        if (!anyEmpCompleted) return false
   424→      }
   425→      return true
   426→    }
   427→
   428→    // Check if the CURRENT USER has completed step 1 individually (EmployeeProgress)
   429→    // This is crucial for unlocking step 2 for an employee who passed the exam
   430→    // even if not ALL employees in the zone have completed it yet
   431→    const hasUserCompletedStep1 = (): boolean => {
   432→      if (!currentZone) return false
   433→      const userStep = employeeProgress.find(ep =>
   434→        ep.sStep === sStep &&
   435→        ep.miniStep === 1 &&
   436→        ep.zoneId === currentZone.id &&
   437→        ep.userId === currentUser.id &&
   438→        ep.completed
   439→      )
   440→      return !!userStep
   441→    }
   442→
   443→    // Check if already completed at zone level (or project level if no zone)
   444→    const isStepCompleted = (): boolean => isStepCompletedAt(sStep, miniStep)
   445→
   446→    // Check if the CURRENT USER has completed this individual step (EmployeeProgress)
   447→    // For step 1 (formación) and step 4 (autoevaluación), completion is tracked per employee
   448→    const hasUserCompletedIndividualStep = (): boolean => {
   449→      if (!currentZone) return false
   450→      if (miniStep !== 1 && miniStep !== 4) return false // Only individual steps
   451→      const userStep = employeeProgress.find(ep =>
   452→        ep.sStep === sStep &&
   453→        ep.miniStep === miniStep &&
   454→        ep.zoneId === currentZone.id &&
   455→        ep.userId === currentUser.id &&
   456→        ep.completed
   457→      )
   458→      return !!userStep
   459→    }
   460→
   461→    // Combined: step is "done" if either zone-level completed OR user completed individually
   462→    const isStepDoneForUser = (): boolean => isStepCompleted() || hasUserCompletedIndividualStep()
   463→
   464→    // Check if step is completed by ANYONE (zone or any employee)
   465→    // This is crucial: ALL users must see completed steps in green,
   466→    // even if they don't have permission to enter them (e.g., auditors viewing steps 1-4)
   467→    const isStepCompletedByAnyone = (): boolean => {
   468→      if (isStepCompleted()) return true
   469→      if (!currentZone) {
   470→        return employeeProgress.some(ep =>
   471→          ep.sStep === sStep &&
   472→          ep.miniStep === miniStep &&
   473→          ep.completed
   474→        )
   475→      }
   476→      return employeeProgress.some(ep =>
   477→        ep.sStep === sStep &&
   478→        ep.miniStep === miniStep &&
   479→        ep.zoneId === currentZone.id &&
   480→        ep.completed
   481→      )
   482→    }
   483→
   484→    // Helper: check if steps 1-4 are all completed for this S-step
   485→    // Uses BOTH zone-level progress AND any employee progress,
   486→    // so auditors can access step 5 when employees have completed 1-4
   487→    const areSteps1to4Completed = (): boolean => {
   488→      for (let ms = 1; ms <= 4; ms++) {
   489→        const zoneCompleted = isStepCompletedAt(sStep, ms)
   490→        if (zoneCompleted) continue
   491→        // Check if ANY employee has completed this step individually
   492→        const anyEmpCompleted = currentZone
   493→          ? employeeProgress.some(ep =>
   494→              ep.sStep === sStep &&
   495→              ep.miniStep === ms &&
   496→              ep.zoneId === currentZone.id &&
   497→              ep.completed
   498→            )
   499→          : employeeProgress.some(ep =>
   500→              ep.sStep === sStep &&
   501→              ep.miniStep === ms &&
   502→              ep.completed
   503→            )
   504→        if (!anyEmpCompleted) return false
   505→      }
   506→      return true
   507→    }
   508→
   509→    // ═══ INTER-S PROGRESSION CHECK ═══
   510→    // S1 is always available. S2 requires S1 completed. S3 requires S2. Etc.
   511→    // This enforces the 5S methodology sequence: Clasificar → Ordenar → Limpiar → Estandarizar → Mantener
   512→    const isPreviousSCompleted = (): boolean => {
   513→      if (sStep === 1) return true // S1 has no prerequisite
   514→      return isSCompleted(sStep - 1)
   515→    }
   516→
   517→    // ═══ INTRA-S PROGRESSION CHECK ═══
   518→    // Step 1: always available (if you have permission AND previous S is done)
   519→    // Steps 2-4: need the immediately previous step completed
   520→    // Step 5: needs ALL steps 1-4 completed
   521→    // IMPORTANT: Completion checks include BOTH zone-level AND any employee progress,
   522→    // so auditors/responsables can see steps as available when employees completed them
   523→    const isPreviousStepCompleted = (): boolean => {
   524→      if (miniStep === 1) return true
   525→      if (miniStep === 5) return areSteps1to4Completed()
   526→      // Steps 2, 3, 4: need previous step completed at zone level
   527→      if (isStepCompletedAt(sStep, miniStep - 1)) return true
   528→      // For step 2: current user completed step 1 individually (exam passed)
   529→      if (miniStep === 2 && hasUserCompletedStep1()) return true
   530→      // Also check: ANY employee completed previous step (for auditors/responsables)
   531→      const anyEmpCompletedPrev = currentZone
   532→        ? employeeProgress.some(ep =>
   533→            ep.sStep === sStep &&
   534→            ep.miniStep === miniStep - 1 &&
   535→            ep.zoneId === currentZone.id &&
   536→            ep.completed
   537→          )
   538→        : employeeProgress.some(ep =>
   539→            ep.sStep === sStep &&
   540→            ep.miniStep === miniStep - 1 &&
   541→            ep.completed
   542→          )
   543→      if (anyEmpCompletedPrev) return true
   544→      return false
   545→    }
   546→
   547→    // ── If already completed (zone-level OR user's individual), VERIFY COHERENCY ──
   548→    if (isStepDoneForUser()) {
   549→      const isChainCoherent = (): boolean => {
   550→        // First check inter-S: previous S must be completed
   551→        if (!isPreviousSCompleted()) return false
   552→        // Then check intra-S chain — reuse same logic as isPreviousStepCompleted
   553→        if (miniStep === 1) return true // Step 1 is always coherent if previous S is done
   554→        if (miniStep === 5) return areSteps1to4Completed()
   555→        if (isStepCompletedAt(sStep, miniStep - 1)) return true
   556→        if (miniStep === 2 && hasUserCompletedStep1()) return true
   557→        // Also check any employee completed previous step
   558→        const anyEmpPrev = currentZone
   559→          ? employeeProgress.some(ep =>
   560→              ep.sStep === sStep &&
   561→              ep.miniStep === miniStep - 1 &&
   562→              ep.zoneId === currentZone.id &&
   563→              ep.completed
   564→            )
   565→          : employeeProgress.some(ep =>
   566→              ep.sStep === sStep &&
   567→              ep.miniStep === miniStep - 1 &&
   568→              ep.completed
   569→            )
   570→        if (anyEmpPrev) return true
   571→        return false
   572→      }
   573→
   574→      if (isChainCoherent()) {
   575→        if (canPerformStep || canSkipSteps) return 'completed'
   576→        if (canViewStep) return 'completed_viewonly'
   577→        return 'locked'
   578→      }
   579→      // Chain is incoherent: treat as NOT completed — fall through
   580→    }
   581→
   582→    // ── Step completed by ANYONE (zone or employee) → show as green for ALL users ──
   583→    // Even users without permission must see completed steps in green
   584→    if (isStepCompletedByAnyone()) {
   585→      if (canPerformStep || canSkipSteps) return 'completed'
   586→      // Everyone else sees it as completed_viewonly (green, read-only)
   587→      return 'completed_viewonly'
   588→    }
   589→
   590→    // ── Admin with lock open: skip all checks ──
   591→    if (skipLocks) return 'available'
   592→
   593→    // ── No permission at all = locked ──
   594→    if (!canViewStep && !canPerformStep) return 'locked'
   595→
   596→    // ── Has perform permission (a1) → check progression ──
   597→    if (canPerformStep) {
   598→      // INTER-S: can't start this S until previous S is completed
   599→      if (!isPreviousSCompleted()) return 'locked'
   600→      // INTRA-S: can't enter step until previous step is completed
   601→      if (!isPreviousStepCompleted()) return 'locked'
   602→      return 'available'
   603→    }
   604→
   605→    // ── Has view permission only (a0) → can see but CANNOT enter ──
   606→    if (canViewStep) return 'locked'
   607→
   608→    return 'locked'
   609→  },
   610→
   611→  isZoneMiniStepComplete: (sStep, miniStep, zoneId) => {
   612→    const { progress, employeeProgress, currentProject } = get()
   613→    if (!currentProject) return false
   614→
   615→    if (ZONE_MINI_STEPS.includes(miniStep)) {
   616→      // Zone step: check if the zone's Progress record is completed
   617→      const zoneStep = progress.find(p =>
   618→        p.sStep === sStep &&
   619→        p.miniStep === miniStep &&
   620→        p.zoneId === zoneId
   621→      )
   622→      return zoneStep?.completed ?? false
   623→    }
   624→
   625→    if (INDIVIDUAL_MINI_STEPS.includes(miniStep)) {
   626→      // Individual step (1 only): ALL employees in the zone must have completed
   627→      // The zone-level Progress record for step 1 is only marked completed when ALL employees pass
   628→      const zoneStep = progress.find(p =>
   629→        p.sStep === sStep &&
   630→        p.miniStep === miniStep &&
   631→        (p.zoneId === zoneId || p.zoneId === null) &&
   632→        p.completed
   633→      )
   634→      return !!zoneStep
   635→    }
   636→
   637→    return false
   638→  },
   639→
   640→  isQuesitoEarned: (sStep) => {
   641→    const { progress, employeeProgress, currentZone } = get()
   642→    const zoneId = currentZone?.id || null
   643→
   644→    // Check ALL 5 mini-steps are completed — each one individually
   645→    // For zone-level steps (2,3,5): check progress table
   646→    // For individual steps (1,4): also check employeeProgress as fallback
   647→    for (let miniStep = 1; miniStep <= 5; miniStep++) {
   648→      // Check zone-level progress first
   649→      const zoneStep = progress.find(p =>
   650→        p.sStep === sStep &&
   651→        p.miniStep === miniStep &&
   652→        (p.zoneId === zoneId || p.zoneId === null) &&
   653→        p.completed
   654→      )
   655→      if (zoneStep) continue
   656→
   657→      // Also check employeeProgress for this step (e.g., individual formación step 1)
   658→      // This ensures the quesito is earned even when not all employees have completed step 1,
   659→      // but the auditor was able to pass step 5 (audit) because areSteps1to4Completed returned true
   660→      const anyEmpCompleted = currentZone
   661→        ? employeeProgress.some(ep =>
   662→            ep.sStep === sStep &&
   663→            ep.miniStep === miniStep &&
   664→            ep.zoneId === currentZone.id &&
   665→            ep.completed
   666→          )
   667→        : employeeProgress.some(ep =>
   668→            ep.sStep === sStep &&
   669→            ep.miniStep === miniStep &&
   670→            ep.completed
   671→          )
   672→      if (anyEmpCompleted) continue
   673→
   674→      return false
   675→    }
   676→    return true
   677→  },
   678→
   679→  is5SCompleted: () => {
   680→    for (let i = 1; i <= 5; i++) {
   681→      if (!get().isQuesitoEarned(i)) return false
   682→    }
   683→    return true
   684→  },
   685→
   686→  getCompletedCount: () => {
   687→    const { progress, currentZone } = get()
   688→
   689→    if (currentZone) {
   690→      const completedMiniSteps = progress.filter(p =>
   691→        (p.zoneId === currentZone.id || p.zoneId === null) &&
   692→        p.completed
   693→      ).length
   694→      let completedSSteps = 0
   695→      for (let i = 1; i <= 5; i++) {
   696→        if (get().isQuesitoEarned(i)) completedSSteps++
   697→      }
   698→      return { sSteps: completedSSteps, miniSteps: completedMiniSteps, total: 25 }
   699→    }
   700→
   701→    const completedMiniSteps = progress.filter(p => p.completed).length
   702→    let completedSSteps = 0
   703→    for (let i = 1; i <= 5; i++) {
   704→      if (get().isQuesitoEarned(i)) completedSSteps++
   705→    }
   706→    return { sSteps: completedSSteps, miniSteps: completedMiniSteps, total: 25 }
   707→  },

   710→  login: async (email: string, password: string) => {
   711→    try {
   712→      set({ isLoginLoading: true, authError: null })
   713→      const res = await fetch('/api/auth', {
   714→        method: 'POST',
   715→        headers: { 'Content-Type': 'application/json' },
   716→        body: JSON.stringify({ email, password }),
   717→      })
   718→      const data = await res.json()
   719→
   720→      if (!res.ok) {
   721→        const errorMsg = data.error || 'Email o contraseña incorrectos'
   722→        set({ isLoginLoading: false, authError: errorMsg })
   723→        return false
   724→      }
   725→
   726→      if (!data.user) {
   727→        set({ isLoginLoading: false, authError: 'Error inesperado al iniciar sesión' })
   728→        return false
   729→      }
   730→
   731→      set({ currentUser: data.user, isLoginLoading: false, authError: null })
   732→
   733→      // Load permissions after login
   734→      try {
   735→        await get().fetchPermissions()
   736→      } catch (e) {
   737→        console.error('Error loading permissions after login:', e)
   738→      }
   739→
   740→      // Gestor (dueño de la app) goes directly to management panel
   741→      if (data.user.role === 'gestor') {
   742→        set({ authView: 'board', activeTab: 'gestion' })
   743→        try {
   744→          await get().fetchCompanies()
   745→        } catch (e) {
   746→          console.error('Error fetching companies after login:', e)
   747→        }
   748→        return true
   749→      }
   750→
   751→      // Check for projects after login
   752→      try {
   753→        await get().fetchProjects()
   754→      } catch (e) {
   755→        console.error('Error fetching projects after login:', e)
   756→      }
   757→      const { projects } = get()
   758→
   759→      if (projects.length > 0) {
   760→        set({ currentProject: projects[0], authView: 'board' })
   761→        // Fetch user's assigned zones after login
   762→        try {
   763→          await get().fetchUserZones()
   764→        } catch (e) {
   765→          console.error('Error fetching user zones after login:', e)
   766→        }
   767→      } else {
   768→        // Only admin can create projects via setup wizard
   769→        // Non-admin users see a waiting screen
   770→        if (data.user.role === 'admin') {
   771→          set({ authView: 'setup' })
   772→        } else {
   773→          set({ authView: 'no_projects' })
   774→        }
   775→      }
   776→
   777→      return true
   778→    } catch (error) {
   779→      console.error('Login error:', error)
   780→      set({ isLoginLoading: false, authError: 'Error de conexión. Inténtalo de nuevo.' })
   781→      return false
   782→    }
   783→  },
