   364→// VISUAL EDITOR: ChecklistEditor (autoevaluacion / auditoria)
   365→// ═══════════════════════════════════════════════════════
   366→interface ChecklistSection {
   367→  id: string
   368→  title: string
   369→  items: { id: string; description: string; hasOther: boolean }[]
   370→}
   371→
   372→function ChecklistEditor({ content, onChange }: { content: string; onChange: (v: string) => void }) {
   373→  let parsed: { sections: ChecklistSection[] }
   374→  try {
   375→    parsed = JSON.parse(content)
   376→  } catch {
   377→    return (
   378→      <div className="p-4 bg-red-50 border border-red-200 rounded-lg text-sm text-red-600">
   379→        El JSON no es válido. Corrígelo en modo JSON o carga el contenido por defecto.
   380→      </div>
   381→    )
   382→  }
   383→
   384→  const sections = parsed.sections || []
   385→  const sPrefix = String(content.match(/"id"\s*:\s*"(\d+)\./)?.[1] || '1')
   386→
   387→  const update = (newSections: ChecklistSection[]) => {
   388→    onChange(JSON.stringify({ ...parsed, sections: newSections }, null, 2))
   389→  }
   390→
   391→  const addSection = () => {
   392→    const newId = `${sPrefix}.${sections.length + 1}`
   393→    update([...sections, { id: newId, title: 'Nueva Sección', items: [] }])
   394→  }
   395→
   396→  const removeSection = (idx: number) => {
   397→    update(sections.filter((_, i) => i !== idx))
   398→  }
   399→
   400→  const updateSection = (idx: number, field: 'id' | 'title', value: string) => {
   401→    const updated = [...sections]
   402→    updated[idx] = { ...updated[idx], [field]: value }
   403→    update(updated)
   404→  }
   405→
   406→  const moveSection = (idx: number, dir: -1 | 1) => {
   407→    const target = idx + dir
   408→    if (target < 0 || target >= sections.length) return
   409→    const updated = [...sections]
   410→    ;[updated[idx], updated[target]] = [updated[target], updated[idx]]
   411→    update(updated)
   412→  }
   413→
   414→  const addItem = (sectionIdx: number) => {
   415→    const sec = sections[sectionIdx]
   416→    const newItemId = `${sec.id}.${sec.items.length + 1}`
   417→    const updated = [...sections]
   418→    updated[sectionIdx] = {
   419→      ...updated[sectionIdx],
   420→      items: [...updated[sectionIdx].items, { id: newItemId, description: '', hasOther: false }]
   421→    }
   422→    update(updated)
   423→  }
   424→
   425→  const removeItem = (sectionIdx: number, itemIdx: number) => {
   426→    const updated = [...sections]
   427→    updated[sectionIdx] = {
   428→      ...updated[sectionIdx],
   429→      items: updated[sectionIdx].items.filter((_, i) => i !== itemIdx)
   430→    }
   431→    update(updated)
   432→  }
   433→
   434→  const updateItem = (sectionIdx: number, itemIdx: number, field: string, value: string | boolean) => {
   435→    const updated = [...sections]
   436→    updated[sectionIdx] = {
   437→      ...updated[sectionIdx],
   438→      items: updated[sectionIdx].items.map((item, i) =>
   439→        i === itemIdx ? { ...item, [field]: value } : item
   440→      )
   441→    }
   442→    update(updated)
   443→  }
   444→
   445→  const moveItem = (sectionIdx: number, itemIdx: number, dir: -1 | 1) => {
   446→    const items = sections[sectionIdx].items
   447→    const target = itemIdx + dir
   448→    if (target < 0 || target >= items.length) return
   449→    const updated = [...sections]
   450→    const newItems = [...updated[sectionIdx].items]
   451→    ;[newItems[itemIdx], newItems[target]] = [newItems[target], newItems[itemIdx]]
   452→    updated[sectionIdx] = { ...updated[sectionIdx], items: newItems }
   453→    update(updated)
   454→  }
   455→
   456→  return (
   457→    <div className="space-y-4">
   458→      {sections.map((section, sIdx) => (
   459→        <div key={sIdx} className="border-2 rounded-lg overflow-hidden bg-white shadow-sm">
   460→          {/* Section header */}
   461→          <div className="flex items-center gap-3 px-4 py-3 bg-gray-50 border-b">
   462→            <div className="flex flex-col gap-0.5">
   463→              <button onClick={() => moveSection(sIdx, -1)} className="text-gray-400 hover:text-gray-600 leading-none" title="Subir sección">
   464→                <ChevronUp className="h-4 w-4" />
   465→              </button>
   466→              <button onClick={() => moveSection(sIdx, 1)} className="text-gray-400 hover:text-gray-600 leading-none" title="Bajar sección">
   467→                <ChevronDown className="h-4 w-4" />
   468→              </button>
   469→            </div>
   470→            <Input
   471→              value={section.id}
   472→              onChange={(e) => updateSection(sIdx, 'id', e.target.value)}
   473→              className="w-20 h-9 text-sm font-mono"
   474→              placeholder="ID"
   475→            />
   476→            <Input
   477→              value={section.title}
   478→              onChange={(e) => updateSection(sIdx, 'title', e.target.value)}
   479→              className="flex-1 h-9 text-base font-semibold"
   480→              placeholder="Título de sección"
   481→            />
   482→            <Button variant="ghost" size="sm" onClick={() => addItem(sIdx)}
   483→              className="h-9 w-9 p-0 text-green-600 hover:text-green-700 hover:bg-green-50" title="Añadir item">
   484→              <Plus className="h-5 w-5" />
   485→            </Button>
   486→            <Button variant="ghost" size="sm" onClick={() => removeSection(sIdx)}
   487→              className="h-9 w-9 p-0 text-red-500 hover:text-red-600 hover:bg-red-50" title="Eliminar sección">
   488→              <Trash2 className="h-4 w-4" />
   489→            </Button>
   490→          </div>
   491→
   492→          {/* Items */}
   493→          <div className="p-3 space-y-2">
   494→            {section.items.length === 0 && (
   495→              <p className="text-sm text-muted-foreground italic px-2 py-2">Sin items. Pulsa + para añadir.</p>
   496→            )}
   497→            {section.items.map((item, iIdx) => (
   498→              <div key={iIdx} className="flex items-center gap-3 group">
   499→                <div className="flex flex-col gap-0.5">
   500→                  <button onClick={() => moveItem(sIdx, iIdx, -1)} className="text-gray-300 hover:text-gray-500 leading-none" title="Subir">
   501→                    <ChevronUp className="h-3.5 w-3.5" />
   502→                  </button>
   503→                  <button onClick={() => moveItem(sIdx, iIdx, 1)} className="text-gray-300 hover:text-gray-500 leading-none" title="Bajar">
   504→                    <ChevronDown className="h-3.5 w-3.5" />
   505→                  </button>
   506→                </div>
   507→                <Input
   508→                  value={item.id}
   509→                  onChange={(e) => updateItem(sIdx, iIdx, 'id', e.target.value)}
   510→                  className="w-20 h-8 text-xs font-mono"
   511→                  placeholder="ID"
   512→                />
   513→                <Input
   514→                  value={item.description}
   515→                  onChange={(e) => updateItem(sIdx, iIdx, 'description', e.target.value)}
   516→                  className="flex-1 h-8 text-sm"
   517→                  placeholder="Descripción del item"
   518→                />
   519→                <label className="flex items-center gap-1.5 text-xs text-muted-foreground whitespace-nowrap cursor-pointer shrink-0">
   520→                  <input
   521→                    type="checkbox"
   522→                    checked={item.hasOther}
   523→                    onChange={(e) => updateItem(sIdx, iIdx, 'hasOther', e.target.checked)}
   524→                    className="rounded border-gray-300 h-4 w-4"
   525→                  />
   526→                  Otros
   527→                </label>
   528→                <Button variant="ghost" size="sm"
   529→                  onClick={() => removeItem(sIdx, iIdx)}
   530→                  className="h-8 w-8 p-0 text-red-400 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity" title="Eliminar">
   531→                  <Trash2 className="h-4 w-4" />
   532→                </Button>
   533→              </div>
   534→            ))}
   535→          </div>
   536→        </div>
   537→      ))}
   538→
   539→      <Button variant="outline" onClick={addSection}
   540→        className="w-full border-dashed border-2 text-green-600 hover:bg-green-50 hover:border-green-400 gap-1 h-10">
   541→        <Plus className="h-5 w-5" />
   542→        Añadir sección
   543→      </Button>
   544→
   545→      <div className="text-sm text-muted-foreground text-center">
   546→        {sections.length} sección(es) · {sections.reduce((s, sec) => s + sec.items.length, 0)} item(s) en total
   547→      </div>
   548→    </div>
   549→  )
   550→}
