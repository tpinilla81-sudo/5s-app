     1→'use client';
     2→
     3→import { useState, useEffect, useMemo, useRef } from 'react';
     4→import {
     5→  Dialog,
     6→  DialogContent,
     7→  DialogHeader,
     8→  DialogTitle,
     9→} from '@/components/ui/dialog';
    10→import { Button } from '@/components/ui/button';
    11→import { Card, CardContent } from '@/components/ui/card';
    12→import { Badge } from '@/components/ui/badge';
    13→import { Input } from '@/components/ui/input';
    14→import { Textarea } from '@/components/ui/textarea';
    15→import { CheckSquare, CheckCircle, XCircle, Camera, ChevronDown, ChevronRight, Maximize2, Minimize2, AlertCircle, Upload, X, Image as ImageIcon } from 'lucide-react';
    16→import { use5SStore } from '@/lib/store';
    17→import {
    18→  S_STEPS,
    19→  SELF_EVAL_THRESHOLD,
    20→} from '@/lib/5s-constants';
    21→import type { AuditSection, AuditItemResult } from '@/lib/5s-constants';
    22→import { useChecklistTemplate } from '@/lib/checklist-templates';
    23→
    24→interface AutoevaluacionModalProps {
    25→  open: boolean;
    26→  onClose: () => void;
    27→  sStep: number;
    28→  miniStep: number;
    29→}
    30→
    31→
    32→
    33→export default function AutoevaluacionModal({ open, onClose, sStep, miniStep }: AutoevaluacionModalProps) {
    34→  const { fetchProgress, currentUser, adminFreeNavigation, currentProject, currentZone, canPerform, canView, hasPermission } = use5SStore();
    35→  const sStepData = S_STEPS.find(s => s.id === sStep);
    36→  const canSkipSteps = hasPermission('skip_steps');
    37→  const canPerformStep = canPerform(sStep, miniStep);
    38→  const canViewStep = canView(sStep, miniStep);
    39→  // Permission-driven: read-only if no execute perm OR if candado closed for skip_steps users
    40→  const isReadOnly = !canPerformStep || (canSkipSteps && !adminFreeNavigation);
    41→  const canPerformAutoeval = canPerformStep;
    42→
    43→  const [isFullscreen, setIsFullscreen] = useState(true);
    44→  const [results, setResults] = useState<Record<string, AuditItemResult>>({});
    45→  const [expandedSections, setExpandedSections] = useState<Record<string, boolean>>({});
    46→  const [observaciones, setObservaciones] = useState('');
    47→  const [isSubmitting, setIsSubmitting] = useState(false);
    48→  const [isCompleted, setIsCompleted] = useState(false);
    49→  const [finalScore, setFinalScore] = useState(0);
    50→  const [notaMinima, setNotaMinima] = useState(70);
    51→  const [autoevalPhotos, setAutoevalPhotos] = useState<{ file: File; preview: string; uploading?: boolean; serverUrl?: string }[]>([]);
    52→  const [isUploadingPhotos, setIsUploadingPhotos] = useState(false);
    53→  const photoInputRef = useRef<HTMLInputElement>(null);
    54→
    55→  // Load template from API (uses board config if zone has one)
    56→  const { sections, isLoading: isLoadingTemplate, notaMinima: templateNotaMinima } = useChecklistTemplate('autoevaluacion', sStep, open, currentZone?.boardConfigId);
    57→
    58→  // Apply template notaMinima when loaded
    59→  useEffect(() => {
    60→    if (templateNotaMinima !== null) setNotaMinima(templateNotaMinima);
    61→  }, [templateNotaMinima]);
    62→
    63→
    64→
    65→  // Fetch dynamic threshold (overrides template if present)
    66→  useEffect(() => {
    67→    if (open && currentProject?.id) {
    68→      const fetchThreshold = async () => {
    69→        try {
    70→          const params = new URLSearchParams({ projectId: currentProject.id });
    71→          if (currentZone?.id) params.set('zoneId', currentZone.id);
    72→          const res = await fetch(`/api/audit-targets?${params}`);
    73→          const json = await res.json();
    74→          if (json.success && json.data) {
    75→            const zoneTarget = json.data.find((t: any) => t.sStep === sStep && t.miniStep === 4 && t.zoneId === currentZone?.id);
    76→            const projectTarget = json.data.find((t: any) => t.sStep === sStep && t.miniStep === 4 && t.zoneId === null);
    77→            const target = zoneTarget || projectTarget;
    78→            if (target?.notaMinima) setNotaMinima(target.notaMinima);
    79→          }
    80→        } catch (e) { console.error('Error fetching threshold:', e); }
    81→      };
    82→      fetchThreshold();
    83→    }
    84→  }, [open, sStep, currentProject?.id, currentZone?.id]);
    85→
    86→  // Initialize expanded sections
    87→  useEffect(() => {
    88→    if (open && sections.length > 0) {
    89→      const expanded: Record<string, boolean> = {};
    90→      sections.forEach(s => { expanded[s.id] = true; });
    91→      setExpandedSections(expanded);
    92→      setResults({});
    93→      setObservaciones('');
    94→      setIsCompleted(false);
    95→      setFinalScore(0);
    96→      setAutoevalPhotos([]);
    97→    }
    98→  }, [open, sStep, sections]);
    99→
   100→  const totalItems = sections.reduce((sum, s) => sum + s.items.length, 0) || 26;
   101→
   102→  const scoring = useMemo(() => {
   103→    const allResults = Object.values(results);
   104→    const okCount = allResults.filter(r => r.status === 'ok').length;
   105→    const nokCount = allResults.filter(r => r.status === 'nok').length;
   106→    const answeredCount = okCount + nokCount;
   107→    const scorePercent = totalItems > 0 ? Math.min(Math.round((okCount / totalItems) * 100), 100) : 0;
   108→    return { okCount, nokCount, answeredCount, scorePercent };
   109→  }, [results, totalItems]);
   110→
   111→  // Check that all NOK items have hallazgo and mejora filled
   112→  const nokItems = Object.values(results).filter(r => r.status === 'nok');
   113→  const allNokCompleted = nokItems.length === 0 || nokItems.every(r => (r.hallazgo || '').trim() !== '' && (r.mejora || '').trim() !== '');
   114→
   115→  const passed = scoring.scorePercent >= notaMinima;
   116→  const canSubmit = canPerformAutoeval && scoring.answeredCount > 0 && allNokCompleted;
   117→
   118→  const handlePhotoSelect = async (e: React.ChangeEvent<HTMLInputElement>) => {
   119→    const files = e.target.files;
   120→    if (!files) return;
   121→    for (const file of Array.from(files)) {
   122→      const preview = URL.createObjectURL(file);
   123→      setAutoevalPhotos(prev => [...prev, { file, preview }]);
   124→    }
   125→    // Reset input
   126→    if (photoInputRef.current) photoInputRef.current.value = '';
   127→  };
   128→
   129→  const removeAutoevalPhoto = (index: number) => {
   130→    setAutoevalPhotos(prev => {
   131→      const photo = prev[index];
   132→      if (photo.preview) URL.revokeObjectURL(photo.preview);
   133→      return prev.filter((_, i) => i !== index);
   134→    });
   135→  };
   136→
   137→  const toggleSection = (sectionId: string) => {
   138→    setExpandedSections(prev => ({ ...prev, [sectionId]: !prev[sectionId] }));
   139→  };
   140→
   141→  const setItemStatus = (itemId: string, status: 'ok' | 'nok' | 'na') => {
   142→    setResults(prev => ({
   143→      ...prev,
   144→      [itemId]: { ...prev[itemId], itemId, status },
   145→    }));
   146→  };
   147→
   148→  const setItemField = (itemId: string, field: 'hallazgo' | 'mejora' | 'otherText', value: string) => {
   149→    setResults(prev => ({
   150→      ...prev,
   151→      [itemId]: { ...prev[itemId], itemId, [field]: value },
   152→    }));
   153→  };
   154→
   155→  const handleSubmit = async () => {
   156→    if (!canSubmit) return;
   157→    if (!canPerformAutoeval) return; // Only responsable/admin for S4, or any employee for S1/S2/S3/S5
   158→    setIsSubmitting(true);
   159→    try {
   160→      // Only mark as completed if score meets notaMinima threshold
   161→      // If not passed, still save results but step stays available for retry
   162→      const res = await fetch(`/api/progress/step?sStep=${sStep}&miniStep=${miniStep}`, {
   163→        method: 'PUT',
   164→        headers: { 'Content-Type': 'application/json' },
   165→        body: JSON.stringify({
   166→          completed: passed,
   167→          score: scoring.scorePercent,
   168→          notes: JSON.stringify({
   169→            type: 'autoevaluacion',
   170→            passed,
   171→            notaMinima,
   172→            results: Object.values(results),
   173→            observaciones,
   174→          }),
   175→          projectId: currentProject?.id,
   176→          zoneId: currentZone?.id || null,
   177→        }),
   178→      });
   179→      const json = await res.json();
   180→      if (json.success) {
   181→        setIsCompleted(true);
   182→        setFinalScore(scoring.scorePercent);
   183→        await fetchProgress();
   184→
   185→        // ─── Create EmployeeProgress record for individual step 4 (autoevaluación) ───
   186→        // Step 4 is individual for S1/S2/S3/S5 (done by employees) and for S4 (done by responsable)
   187→        // We need to track individual completion so the gating system unlocks step 5
   188→        if (passed && currentUser?.id && currentProject?.id && currentZone?.id) {
   189→          try {
   190→            await fetch('/api/employee-progress', {
   191→              method: 'POST',
   192→              headers: { 'Content-Type': 'application/json' },
   193→              body: JSON.stringify({
   194→                sStep,
   195→                miniStep: 4,
   196→                completed: true,
   197→                score: scoring.scorePercent,
   198→                projectId: currentProject.id,
   199→                zoneId: currentZone.id,
   200→                userId: currentUser.id,
   201→              }),
   202→            });
   203→            // Also refresh employee progress so step 5 unlocks immediately
   204→            const { fetchEmployeeProgress } = use5SStore.getState();
   205→            await fetchEmployeeProgress(currentProject.id, currentZone.id);
   206→          } catch (empErr) {
   207→            console.error('Error creating employee progress for autoeval:', empErr);
   208→          }
   209→        }
   210→
   211→        // ─── Upload photos to library with traceability ───
   212→        if (autoevalPhotos.length > 0) {
   213→          setIsUploadingPhotos(true);
   214→          for (let idx = 0; idx < autoevalPhotos.length; idx++) {
   215→            const photo = autoevalPhotos[idx];
   216→            try {
   217→              // Step 1: Upload the file to get a server URL
   218→              const formData = new FormData();
   219→              formData.append('file', photo.file);
   220→              formData.append('filename', `S${sStep}_autoeval_${currentZone?.name || 'zona'}_${idx + 1}_${Date.now()}.jpg`);
   221→
   222→              const uploadRes = await fetch('/api/upload', {
   223→                method: 'POST',
   224→                body: formData,
   225→              });
   226→              const uploadData = await uploadRes.json();
   227→
   228→              if (uploadData.success && uploadData.url) {
   229→                // Step 2: Save to photo library with full traceability
   230→                await fetch('/api/photo-library', {
   231→                  method: 'POST',
   232→                  headers: { 'Content-Type': 'application/json' },
   233→                  body: JSON.stringify({
   234→                    sStep,
   235→                    miniStep: 4,
   236→                    title: `Autoeval S${sStep} - Foto ${idx + 1}`,
   237→                    description: `${sStepData?.japaneseName || 'S' + sStep} - ${currentZone?.name || 'Zona'} - Paso 4 Autoevaluación - Subida por ${currentUser?.name || 'Usuario'}`,
   238→                    photoUrl: uploadData.url,
   239→                    photoType: 'hallazgo',
   240→                    category: `paso4_s${sStep}`,
   241→                    tags: JSON.stringify([`S${sStep}`, sStepData?.japaneseName || '', currentZone?.name || '', 'paso4', 'autoevaluacion', 'hallazgo']),
   242→                    projectId: currentProject?.id,
   243→                    zoneId: currentZone?.id || null,
   244→                    uploadedBy: currentUser?.id || null,
   245→                  }),
   246→                });
   247→              }
   248→            } catch (photoErr) {
   249→              console.error('Error uploading autoeval photo:', photoErr);
   250→            }
   251→          }
   252→          setIsUploadingPhotos(false);
   253→        }
   254→
   255→        // ─── Create Action Items for NOK (disfunciones) ───
   256→        const nokResults = Object.values(results).filter((r: any) => r.status === 'nok');
   257→        for (const nok of nokResults) {
   258→          if (!nok.hallazgo && !nok.mejora) continue; // Skip items without description
   259→          try {
   260→            await fetch('/api/actions', {
   261→              method: 'POST',
   262→              headers: { 'Content-Type': 'application/json' },
   263→              body: JSON.stringify({
   264→                sStep,
   265→                miniStep: 4,
   266→                itemId: nok.itemId,
   267→                itemDescription: `Disfunción detectada en autoevaluación: ${nok.itemId}`,
   268→                hallazgo: nok.hallazgo || nok.itemId,
   269→                mejora: nok.mejora || '',
   270→                responsable: null,
   271→                prioridad: 'media',
   272→                estado: 'abierta',
   273→                source: 'autoevaluacion',
   274→                auditor: null,
   275→                projectId: currentProject?.id,
   276→                zoneId: currentZone?.id || null,
   277→              }),
   278→            });
   279→          } catch (actionError) {
   280→            console.error('Error creating action item from autoevaluación:', actionError);
   281→          }
   282→        }
   283→
   284→        // ─── Notify responsables of NOK disfunciones ───
   285→        if (nokResults.length > 0 && currentProject?.id && currentZone?.id) {
   286→          try {
   287→            const membersRes = await fetch(`/api/projects/${currentProject.id}/members`);
   288→            const membersData = await membersRes.json();
   289→            const allMembers = membersData?.members || [];
   290→
   291→            const sStepData = S_STEPS.find(s => s.id === sStep);
   292→            const disfuncionMessage = `Se han detectado ${nokResults.length} disfunción(es) en la autoevaluación de S${sStep} (${sStepData?.japaneseName || ''}) en la zona "${currentZone.name}". Revisa el Plan de Acción.`;
   293→
   294→            // Notify zone responsable
   295→            const responsableIds = new Set<string>();
   296→            if (currentZone.responsableId) responsableIds.add(currentZone.responsableId);
   297→            const responsables = allMembers.filter((m: any) => m.role === 'responsable');
   298→            for (const resp of responsables) responsableIds.add(resp.userId);
   299→
   300→            for (const respId of responsableIds) {
   301→              await fetch('/api/notifications', {
   302→                method: 'POST',
   303→                headers: { 'Content-Type': 'application/json' },
   304→                body: JSON.stringify({
   305→                  userId: respId,
   306→                  type: 'disfuncion',
   307→                  title: `Disfunciones detectadas: S${sStep} — ${sStepData?.japaneseName || ''}`,
   308→                  message: disfuncionMessage,
   309→                  sStep,
   310→                  zoneId: currentZone.id,
   311→                  projectId: currentProject.id,
   312→                }),
   313→              });
   314→            }
   315→          } catch (notifError) {
   316→            console.error('Error notifying responsables of disfunciones:', notifError);
   317→          }
   318→        }
   319→
   320→        // Check if Steps 1-4 are now all completed for this S-step in the zone
   321→        // If so, notify auditor(s) that they can perform Step 5
   322→        if (currentProject?.id && currentZone?.id) {
   323→          try {
   324→            // Fetch all progress for this zone and S-step
   325→            const progRes = await fetch(`/api/progress?projectId=${currentProject.id}`);
   326→            const progData = await progRes.json();
   327→            const allProgress = progData?.data || [];
   328→
   329→            // Also fetch employee progress for individual steps (formación = step 1)
   330→            const empProgRes = await fetch(`/api/employee-progress?projectId=${currentProject.id}&zoneId=${currentZone.id}`);
   331→            const empProgData = await empProgRes.json();
   332→            const allEmpProgress = empProgData?.data || [];
   333→
   334→            // Check steps 1-4 completed (both zone-level progress AND employee progress)
   335→            let allStepsCompleted = true;
   336→            for (let ms = 1; ms <= 4; ms++) {
   337→              const zoneStep = allProgress.find((p: any) =>
   338→                p.sStep === sStep &&
   339→                p.miniStep === ms &&
   340→                (p.zoneId === currentZone.id || p.zoneId === null) &&
   341→                p.completed
   342→              );
   343→              if (zoneStep) continue; // Zone-level completed
   344→
   345→              // Also check employee progress for individual steps
   346→              const empStep = allEmpProgress.some((ep: any) =>
   347→                ep.sStep === sStep &&
   348→                ep.miniStep === ms &&
   349→                ep.zoneId === currentZone.id &&
   350→                ep.completed
   351→              );
   352→              if (empStep) continue; // Some employee completed this step
   353→
   354→              allStepsCompleted = false;
   355→              break;
   356→            }
   357→
   358→            if (allStepsCompleted) {
   359→              // Find auditor and responsable users for this project and notify them
   360→              const membersRes = await fetch(`/api/projects/${currentProject.id}/members`);
   361→              const membersData = await membersRes.json();
   362→              const allMembers = membersData?.members || [];
   363→
   364→              const sStepData = S_STEPS.find(s => s.id === sStep);
   365→              const notifMessage = `Los pasos 1-4 de S${sStep} (${sStepData?.japaneseName || ''}) en la zona "${currentZone.name}" han sido completados. La auditoría (Paso 5) está lista para realizarse.`;
   366→
   367→              // Notify auditors
   368→              const auditors = allMembers.filter((m: any) => m.role === 'auditor');
   369→              for (const auditor of auditors) {
   370→                await fetch('/api/notifications', {
   371→                  method: 'POST',
   372→                  headers: { 'Content-Type': 'application/json' },
   373→                  body: JSON.stringify({
   374→                    userId: auditor.userId,
   375→                    type: 'audit_ready',
   376→                    title: `Auditoría lista: S${sStep} — ${sStepData?.japaneseName || ''}`,
   377→                    message: notifMessage,
   378→                    sStep,
   379→                    zoneId: currentZone.id,
   380→                    projectId: currentProject.id,
   381→                  }),
   382→                });
   383→              }
   384→
   385→              // Notify responsable of the zone (from zone.responsableId OR from project members)
   386→              const responsableIds = new Set<string>();
   387→              if (currentZone.responsableId) responsableIds.add(currentZone.responsableId);
   388→              const responsables = allMembers.filter((m: any) => m.role === 'responsable');
   389→              for (const resp of responsables) responsableIds.add(resp.userId);
   390→              for (const respId of responsableIds) {
   391→                await fetch('/api/notifications', {
   392→                  method: 'POST',
   393→                  headers: { 'Content-Type': 'application/json' },
   394→                  body: JSON.stringify({
   395→                    userId: respId,
   396→                    type: 'audit_ready',
   397→                    title: `Auditoría lista: S${sStep} — ${sStepData?.japaneseName || ''}`,
   398→                    message: notifMessage,
   399→                    sStep,
   400→                    zoneId: currentZone.id,
   401→                    projectId: currentProject.id,
   402→                  }),
   403→                });
   404→              }
   405→            }
   406→          } catch (notifError) {
   407→            console.error('Error sending notification to auditor:', notifError);
   408→          }
   409→        }
   410→      }
   411→    } catch (error) {
   412→      console.error('Error submitting self-evaluation:', error);
   413→    } finally {
   414→      setIsSubmitting(false);
   415→    }
   416→  };
   417→
   418→  const handleAdminSkip = async () => {
   419→    try {
   420→      const res = await fetch(`/api/progress/step?sStep=${sStep}&miniStep=${miniStep}`, {
   421→        method: 'PUT',
   422→        headers: { 'Content-Type': 'application/json' },
   423→        body: JSON.stringify({ completed: true, score: 100, notes: 'Completado por administrador (skip)', skipMissingTemplate: true, projectId: currentProject?.id, zoneId: currentZone?.id || null }),
   424→      });
   425→      const json = await res.json();
   426→      if (json.success) {
   427→        await fetchProgress();
   428→        onClose();
   429→      }
   430→    } catch (error) {
   431→      console.error('Error admin skip:', error);
   432→    }
   433→  };
   434→
   435→  return (
   436→    <Dialog open={open} onOpenChange={() => onClose()}>
   437→      <DialogContent size={isFullscreen ? "fullscreen" : "xl"} className="flex flex-col overflow-hidden p-0">
   438→        <DialogHeader className="px-6 pt-6 pb-2 flex-shrink-0">
   439→          <DialogTitle className="flex items-center gap-2">
   440→            <CheckSquare className="h-5 w-5" style={{ color: sStepData?.color }} />
   441→            <span>Autoevaluación Interna</span>
   442→            <Badge variant="outline" style={{ borderColor: sStepData?.color, color: sStepData?.color }}>
   443→              {sStepData?.japaneseName} — {sStepData?.spanishName}
   444→            </Badge>
   445→            <button
   446→              onClick={() => setIsFullscreen(!isFullscreen)}
   447→              className="ml-auto p-1 rounded hover:bg-muted transition-colors"
   448→              title={isFullscreen ? "Reducir ventana" : "Pantalla completa"}
   449→            >
   450→              {isFullscreen ? <Minimize2 className="h-4 w-4 text-muted-foreground" /> : <Maximize2 className="h-4 w-4 text-muted-foreground" />}
   451→            </button>
   452→          </DialogTitle>
   453→        </DialogHeader>
   454→
   455→        {canSkipSteps && !isCompleted && (
   456→          <div className="flex items-center gap-2 p-2 mx-6 flex-shrink-0 bg-amber-50 border border-amber-200 rounded-lg">
   457→            <span className="text-xs text-amber-700 font-medium">Modo Admin:</span>
   458→            <Button
   459→              variant="outline"
   460→              size="sm"
   461→              className="text-xs border-amber-300 text-amber-700 hover:bg-amber-100"
   462→              onClick={handleAdminSkip}
   463→            >
   464→              Completar paso sin autoevaluación
   465→            </Button>
   466→          </div>
   467→        )}
   468→
   469→        {isReadOnly && !isCompleted && (
   470→          <div className="flex items-center gap-2 p-2 mx-6 flex-shrink-0 bg-blue-50 border border-blue-200 rounded-lg">
   471→            <span className="text-xs text-blue-700 font-medium">Solo lectura: {canSkipSteps ? 'Activa el candado para poder realizar pasos.' : 'Puedes ver pero no modificar.'}</span>
   472→          </div>
   473→        )}
   474→
   475→        <div className="flex-1 overflow-y-auto px-6 pb-6 min-h-0">
   476→        {isLoadingTemplate ? (
   477→          <div className="flex items-center justify-center py-12">
   478→            <div className="h-8 w-8 border-4 border-green-500 border-t-transparent rounded-full animate-spin" />
   479→          </div>
   480→        ) : sections.length === 0 ? (
   481→          <div className="text-center py-12">
   482→            <AlertCircle className="h-12 w-12 text-amber-500 mx-auto mb-3" />
   483→            <h3 className="text-lg font-bold mb-2">No hay checklist configurado</h3>
   484→            <p className="text-sm text-muted-foreground max-w-md mx-auto">
   485→              No se encontró una plantilla de autoevaluación para S{sStep}. 
   486→              Crea una plantilla en <strong>Administración → Plantillas → Auditoría Interna</strong> 
   487→              o pulsa el botón &quot;Crear plantillas por defecto&quot;.
   488→            </p>
   489→          </div>
   490→        ) : isCompleted ? (
   491→          <div className="text-center py-8">
   492→            {passed ? (
   493→              <>
   494→                <CheckCircle className="h-16 w-16 text-green-500 mx-auto mb-3" />
   495→                <h3 className="text-xl font-bold mb-2 text-green-700">¡Autoevaluación Aprobada!</h3>
   496→              </>
   497→            ) : (
   498→              <>
