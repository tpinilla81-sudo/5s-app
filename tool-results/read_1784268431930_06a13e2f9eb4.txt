   667→/** Calculate total items per S from the actual checklist data */
   668→export function getAuditTotalItems(sStep: number): number {
   669→  const sections = AUDIT_CHECKLISTS[sStep];
   670→  if (!sections) return 0;
   671→  return sections.reduce((sum, section) => sum + section.items.length, 0);
   672→}
