   482→  const handleAttachPhoto = async (itemId: string, file: File, photoType: string) => {

   489→      // Read and compress the image
   490→      const reader = new FileReader();
   491→      const base64Promise = new Promise<string>((resolve) => {
   492→        reader.onloadend = () => resolve(reader.result as string);
   493→        reader.readAsDataURL(file);
   494→      });
   495→      const rawBase64 = await base64Promise;
   496→      const compressed = await compressImage(rawBase64);
   497→
   498→      // Upload to server
   499→      const filename = generatePhotoFilename(currentProject.id, sStep, miniStep, Date.now());
   500→      const uploadFile = base64toFile(compressed, filename);
   501→      const formData = new FormData();
   502→      formData.append('file', uploadFile);
   503→      formData.append('filename', filename);
   504→      formData.append('projectId', currentProject.id);
   505→
   506→      const uploadRes = await fetch('/api/upload', { method: 'POST', body: formData });
   507→      const uploadJson = await uploadRes.json();
   508→
   509→      if (!uploadJson.success || !uploadJson.url) {
   510→        toast.error(`Error al subir foto: ${uploadJson.error || 'Error desconocido'}`);
   511→        return;
   512→      }
   513→
   514→      // Save to PhotoLibrary with inventoryItemId
   515→      const sName = sStepData?.japaneseName || `S${sStep}`;

   830→        const sheetName = workbook.SheetNames[0]; // Use first sheet

   834→        if (rawData.length < 2) {
   835→          toast.error('El archivo está vacío o no tiene datos');
   836→          e.target.value = '';
   837→          return;

   845→            headerIdx = i;
   846→            break;
   847→          }

   851→          // Count non-empty, non-numeric-only cells

   854→            return v !== '' && v !== '0';
   855→          });
