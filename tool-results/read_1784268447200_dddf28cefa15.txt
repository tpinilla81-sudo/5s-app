   470→  // Unified file import: supports both .csv and .xlsx files
   471→  const handleFileImport = async (e: React.ChangeEvent<HTMLInputElement>) => {
   472→    const file = e.target.files?.[0];
   473→    if (!file) return;
   474→
   475→    try {
   476→      const fileName = file.name.toLowerCase();
   477→      let dataRows: string[][] = [];
   478→      let headerRow: string[] = [];
   479→
   480→      if (fileName.endsWith('.xlsx') || fileName.endsWith('.xls')) {
   481→        // Parse Excel file using xlsx library
   482→        const XLSX = await import('xlsx');
   483→        const arrayBuffer = await file.arrayBuffer();
   484→        const workbook = XLSX.read(arrayBuffer, { type: 'array' });
   485→        const sheetName = workbook.SheetNames[0]; // Use first sheet
   486→        const sheet = workbook.Sheets[sheetName];
   487→        const rawData: string[][] = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: '' });
   488→        
   489→        if (rawData.length < 2) {
   490→          toast.error('El archivo está vacío o no tiene datos');
   491→          e.target.value = '';
   492→          return;
   493→        }
   494→
   495→        // Find the header row: look for a row that contains 'Nº' or 'Elemento' or 'Nombre'
   496→        let headerIdx = 0;
   497→        for (let i = 0; i < Math.min(10, rawData.length); i++) {
   498→          const rowStr = rawData[i].map(c => String(c).toLowerCase()).join('|');
   499→          if (rowStr.includes('elemento') || rowStr.includes('nombre') || rowStr.includes('nº') || rowStr.includes('punto')) {
   500→            headerIdx = i;
   501→            break;
   502→          }
   503→        }
   504→        headerRow = rawData[headerIdx].map((h: any) => String(h).trim().toLowerCase());
